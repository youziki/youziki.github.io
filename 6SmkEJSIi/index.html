题解 AT1983 [AGC001E] BBQ Hard

2020-10-30 题解

洛谷传送门

可以把组合数转化成平面上两点距离，然后就可以对每一个点求从其他点出发到它的方案数，再减去自己到自己的方案数

具体而言：直接转化为两点之间距离的话会有(Ai+Aj,Bi+Bj)(A_i + A_j, B_i + B_j)(Ai​+Aj​,Bi​+Bj​)，平移就变成了(−Ai,−Bi)(-A_i,-B_i)(−Ai​,−Bi​)到(Aj,Bj)(A_j, B_j)(Aj​,Bj​)的方案数，所以从f[−Ai][−Bi]f[-A_i][-B_i]f[−Ai​][−Bi​]出发，到正的值，需要加个偏移量

#include <bits/stdc++.h>
using namespace std;
#define int long long

const int MA = 1 << 23;
char buf[MA], *p1 = buf, *p2 = buf;
#define gc() (p1 == p2 && (p2 = (p1 = buf) + fread(buf, 1, MA, stdin), p1 == p2) ? EOF : *p1 ++)

inline int read()
{
	int ff = 0, rr = 1;
	char ch = gc();
	while(! isdigit(ch))
	{
		if(ch == '-') rr = -1;
		ch = gc();
	}
	while(isdigit(ch))
	{
		ff = (ff << 1) + (ff << 3) + (ch ^ 48);
		ch = gc();
	}
	return ff * rr;
}

void print(int x)
{
	if(x < 0) putchar('-'), x = -x;
	if(x > 9) print(x / 10);
	putchar(x % 10 + '0');
}

const int N = 1233333;
const int M = 2020;
const int mod = 1e9 + 7;
const int inv2 = (mod + 1) >> 1;

inline int power(int a, int b)
{
    int res = 1;
    for(; b; b >>= 1, a = a * a % mod)
        if(b & 1) res = res * a % mod;
    return res;
}

int f[M << 1][M << 1], a[N], b[N], ans, mxa, mxb;
int jc[N], inv[N];
bool v[M << 1][M << 1];

int dp(int i, int j)
{
    if(i < - mxa || j < - mxb) return 0;
    if(v[i + M][j + M]) return f[i + M][j + M];
    v[i + M][j + M] = 1;
    return f[i + M][j + M] = (f[i + M][j + M] + dp(i - 1, j) + dp(i, j - 1)) % mod;
}

signed main()
{
    int n = read();
    for(int i = 1; i <= n; ++i)
    {
        a[i] = read(), b[i] = read();
        mxa = max(mxa, a[i]);
        mxb = max(mxb, b[i]);
        ++ f[- a[i] + M][- b[i] + M];
    }

    jc[0] = 1;
    for(int i = 1; i < (M << 2); ++i) jc[i] = jc[i - 1] * i % mod;
	inv[1] = 1;
	for(int i = 2;i < (M << 2);i ++) inv[i] = (mod - mod / i) * inv[mod % i] % mod;
	inv[0] = 1;
	for(int i = 1;i < (M << 2);i ++) inv[i] = inv[i] * inv[i - 1] % mod;
    
    for(int i = 1; i <= n; ++i) ans = (ans + dp(a[i], b[i])) % mod;
    for(int i = 1; i <= n; ++i) ans = (ans - jc[(a[i] + b[i]) << 1] * inv[a[i] << 1] % mod * inv[b[i] << 1] % mod + mod) % mod;

    print(ans * inv2 % mod);

    return 0;
}
