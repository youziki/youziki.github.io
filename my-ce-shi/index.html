#include <bits/stdc++.h>
using namespace std;

const int MA = 1 << 23;
char buf[MA], *p1 = buf, *p2 = buf;
#define gc() (p1 == p2 && (p2 = (p1 = buf) + fread(buf, 1, MA, stdin), p1 == p2) ? EOF : *p1 ++)

inline int read()
{
    int ff = 0, rr = 1;
    char ch = gc();
    while(! isdigit(ch))
    {
        if(ch == '-') rr = -1;
        ch = gc();
    }
    while(isdigit(ch))
    {
        ff = (ff << 1) + (ff << 3) + (ch ^ 48);
        ch = gc();
    }
    return ff * rr;
}

struct Splay
{
    int s[2], p, v;
    int siz;

    inline void init(int _v, int _p)
    {
        siz = 1, v = _v, p = _p;
    }
} _splay[N];

#define _s _splay
#define ls _s[p].s[0]
#define rs _s[p].s[1]

int rt, tot;

inline void pushup(int p)
{
    _s[p].siz = _s[ls].siz + _s[rs].siz + 1;
}

inline void rotate(int x)
{
    int y = _s[x].p, z = _s[y].p;
    int kx = _s[y].s[1] == x, ky = _s[z].s[1] == y;
    _s[z].s[ky] = x, _s[x].p = z;
    _s[y].s[kx] = _s[x].s[kx ^ 1], _s[_s[x].s[kx ^ 1]] = y;
    _s[x].s[kx ^ 1] = y, _s[y].p = x;
    pushup(y), pushup(x);
}

inline void splay(int x, int k)
{
    while(_s[x].p != k)
    {
        int y = _s[x].p, z = _s[y].p;
        if(z != k)
            if((_s[y].s[1] == x) ^ (_s[z].s[1] == y))
                rotate(x);
            else rotate(y);
        rotate(x);
    }
    if(! k) rt = x;
}
