讲题 P4165组队

2020-10-30 题解

传送门

暴力枚举每个minh，暴力对每个人求出一个使这个人可以被选的minv范围，差分表示u，最后求一遍前缀和，最大的数即为选这个为minv时人数最大，最大的人数

#include<bits/stdc++.h>
#define int long long
using namespace std;

const int M=5005;
int n,A,B,C,h[M],hh[M],s[M],hs=1,ans=1;
int cnt[M*2],smax;

signed main()
{
	cin>>n>>A>>B>>C;
	for (int i=1;i<=n;i++)
		cin>>h[i]>>s[i],smax=max(smax,s[i]),hh[i]=h[i];
	sort(hh+1,hh+1+n);
	for (int i=2;i<=n;i++)
		if (hh[i]!=hh[i-1])
			hh[++hs]=hh[i];
	for (int i=1;i<=hs;i++)
	{
		memset(cnt,0,sizeof(cnt));
		for (int j=1;j<=n;j++)
		{
			if (h[j]>=hh[i]&&(A*(h[j]-hh[i]))<=C)
			{
				int tl;
				if (B==0)
					tl=1;
				else
					tl=max(1ll,s[j]-(C-A*(h[j]-hh[i]))/B);
				 
				cnt[tl]++,cnt[s[j]+1]--;
			}
		}
		for (int j=2;j<=smax;j++)
			cnt[j]+=cnt[j-1];
		for (int j=1;j<=n;j++)
			ans=max(ans,cnt[s[j]]);
	}
	cout<<ans;
	return 0;
}
