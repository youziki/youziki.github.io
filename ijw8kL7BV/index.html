题解 AT3621 [ARC084B] Small Multiple

2020-10-30 题解

洛谷传送门

思路很巧，实现也是一个知识点

首先任意一个大于等于kkk的数都可以由k每次+1+ 1+1或×10\times 10×10得到，所以就这样连边，+1+ 1+1连边权为1，×10\times 10×10连边权为0，跑双端队列bfsbfsbfs

实现： 起点是1，跑到%k意义下为0就输出，只要是最短路，这样的结果就一定是最小的，要保证是最短路，就要0边权的从队头进入，1边权的从队尾进入

#include <bits/stdc++.h>
using namespace std;

const int N = 123333;

struct Node
{
    int num, k;
};

int k;
bool vis[N];
deque <Node> q;

inline void bfs()
{
    while(q.size())
    {
        auto x = q.front(); q.pop_front();
        if(vis[x.num]) continue;
        vis[x.num] = 1;
        if(! x.num)
        {
            printf("%d", x.k);
            return;
        }
        q.push_front( {x.num * 10 % k, x.k} );
        q.push_back( {(x.num + 1) % k, x.k + 1});
    }
}

signed main()
{
    scanf("%d", &k);
    q.push_front( {1, 1} );
    bfs();
    
    return 0;
}
