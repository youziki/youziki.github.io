题解 11.13 模拟赛

2020-11-14 比赛

T1 gcd

Solution：

思路真不错！咋想到的呀？

  • 考虑一种暴力：

    答案为从1到i的所有数的gcd求和（区间）

    等价做法：记录从每个位置开始的后缀，每次与每个后缀做gcd，并更新这些后缀的gcd，同时开一个以当前数开始的后缀

  • 优化：

    第二个做法加一个判断：是否是前面那个数（gcd）的倍数，如果是，直接break。还要记录一个前面所有位置开始的后缀的前缀和，如果修改，同时O(1)O(1)O(1)修改这个前缀和，每次加上这个前缀和（修改后）即可

  • 时间复杂度证明：

    每个g[]只会被修改30次，所以总复杂度O(nloga)O(nloga)O(nloga)。也可以换种方式理解，如果有一个位置会一直往前遍历（前面的g[]都相同），那么得到前面这些g[]的时候，全是O(1)的，所以这种情况会被均摊掉

Code：

signed main()
{
	scanf("%lld", &n);
	for (int i = 1, a; i <= n; ++i)
	{
		scanf("%lld", &a);
		g[i] = a, (sum += a) %= mod;
		int j = i - 1;
		while (j && a % g[j])
			(sum -= g[j]) %= mod, g[j] = __gcd(g[j], a), (sum += g[j]) %= mod,
			--j;
		(ans += sum) %= mod;
	}
	printf("%lld", (ans + mod) % mod);
	
	return 0;
}

T2 or_and

Solution：

练习取模的好题！

考场上很快推出式子，写完，然后爆了

long long开少一点都不行，取mod取少一点都不行，注意：n和m的范围是1e18，在运算的时候要取模，但不能先对n和m取模，因为它们还表示数量

Code：

#define int long long

const int mod = 1e9 + 7;

#define P mod

signed main()
{	
	int n, m;
	int ans = 0;
	scanf("%lld%lld", &n, &m);
	if (n < m) swap(n, m);
	
	++ n, ++ m;
	
	int val = 1;
	for (int i = 0; val < n; ++i, val <<= 1)
	{
		int times_n = (val > n / 2) ?
				n - val :
				n / (val << 1) * val % P
					+ ( (n % (val << 1) > val) ? n % (val << 1) - val : 0 ), times_m = 0;
		if (val <= m)
			times_m = (val > m / 2) ?
				m - val :
				m / (val << 1) * val % P
					+ ( (m % (val << 1) > val) ? m % (val << 1) - val : 0 );
		times_n %= P, times_m %= P;
		int t = (
				(times_n * (m % P) % P + times_m * (n % P) % P - times_n * times_m % P) % P
				+ P ) % P;
		ans = (ans + t * (val % P) % P) % P;
	}
	
	printf("%lld", ans);
	
	return 0;
}

T4 beautiful

Solution：

一看就知道是反悔贪心，可惜不会。。。代码很简单的，看看吧，从后往前放右括号是为了保证每一个后缀右括号都不少于左括号

是时候该学习反悔贪心了？！

Code：

#include <bits/stdc++.h>

using namespace std;

#define LL long long

const int N = 300005;

int n, a[N], b[N];
LL sum;

signed main()
{
	scanf("%d", &n);

	for (int i = 1; i <= n; ++i) scanf("%d", a + i), sum += a[i];
	for (int i = 1; i <= n; ++i) scanf("%d", b + i), b[i] = a[i] - b[i];

	priority_queue<int> q;
	q.push(b[n]);
	for (int i = 1; i <= n / 2; ++i)
	{
		int v = q.top(); q.pop();
		sum -= v;
		q.push(b[n - 2 * i + 1]), q.push(b[n - 2 * i]);
	}

	printf("%lld", sum);

	return 0;
}
