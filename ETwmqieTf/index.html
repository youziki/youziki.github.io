P4514 上帝造题的七分钟

Solution：

    1. 区间加肯定是要差分的，强制在线就不能O(n^2)求前缀和，考虑树状数组动态维护前缀和 （不就是二维的嘛）
    1. 修改差分，前缀和表示单点值。要询问区间，就要再用二维前缀和的思想求一遍前缀和做一遍容斥
    1. 实际做起来很麻烦，推一推式子：

    ∑i=1x∑j=1y∑h=1i∑k=1ja[h][k]\sum_{i = 1}^{x} \sum_{j = 1}^{y} \sum_{h = 1}^{i} \sum_{k = 1}^{j} a[h][k]∑i=1x​∑j=1y​∑h=1i​∑k=1j​a[h][k]

    后两个∑\sum∑是单点值，前两个是在(x,y)位置处的二维前缀和，实际询问时就拿这个式子容斥（4个部分）

    考虑每一个d[h][k]]d[h][k]]d[h][k]]的贡献（出现的次数），有下面的式子：

    ∑i=1x∑j=1ya[i][j]×(x−i+1)×(y−j+1)\sum_{i = 1}^{x} \sum_{j = 1}^{y} a[i][j] \times (x - i + 1) \times (y - j + 1)∑i=1x​∑j=1y​a[i][j]×(x−i+1)×(y−j+1)

    整理得：

    ∑i=1x∑j=1ya[i][j]×(x×y+x+y+1)−a[i][j]×i×(y+1)−a[i][j]×j×(x+1)+a[i][j]×i×j\sum_{i = 1}^{x} \sum_{j = 1}^{y} a[i][j] \times (x \times y + x + y + 1) - a[i][j] \times i \times (y + 1) - a[i][j] \times j \times (x + 1) + a[i][j] \times i \times j∑i=1x​∑j=1y​a[i][j]×(x×y+x+y+1)−a[i][j]×i×(y+1)−a[i][j]×j×(x+1)+a[i][j]×i×j

    就等于：

    (x×y+x+y+1)∑i=1x∑j=1ya[i][j]−(y+1)∑i=1x∑j=1ya[i][j]×i−(x+1)∑i=1x∑j=1ya[i][j]×j+∑i=1x∑j=1ya[i][j]×i×j(x \times y + x + y + 1)\sum_{i = 1}^{x} \sum_{j = 1}^{y} a[i][j] - (y + 1) \sum_{i = 1}^{x} \sum_{j = 1}^{y} a[i][j] \times i - (x + 1)\sum_{i = 1}^{x} \sum_{j = 1}^{y} a[i][j] \times j + \sum_{i = 1}^{x} \sum_{j = 1}^{y} a[i][j] \times i \times j(x×y+x+y+1)∑i=1x​∑j=1y​a[i][j]−(y+1)∑i=1x​∑j=1y​a[i][j]×i−(x+1)∑i=1x​∑j=1y​a[i][j]×j+∑i=1x​∑j=1y​a[i][j]×i×j

    于是就需要维护a[i][j],a[i][j]×i,a[i][j]×j,a[i][j]×i×ja[i][j], a[i][j] \times i, a[i][j] \times j, a[i][j] \times i \times ja[i][j],a[i][j]×i,a[i][j]×j,a[i][j]×i×j的前缀和四个数组。这里代码有些容易理解错的点，注意：

    addTadd_TaddT​函数内增量全为k,k∗x,k∗y,k∗x∗yk,k*x,k*y,k*x*yk,k∗x,k∗y,k∗x∗y是因为，我们只在(x, y)（在式子中是(i, j)）位置增加了k，所以后面被影响到的增量是(x, y)为位置的增量，askTask_TaskT​函数内的x、y的含义不同，此时x和y为上面式子里的x和y，是前缀和的系数

    1. 总结一下，二维树状数组的两个核心函数的写法是“板子”的基础，本题是区间修改，所以需要多维护一个差分，两个前缀和加起来就需要推式子了。学习本题应先掌握板子部分，本题的扩展部分就好理解了。
    1. 思考：如果只是区间修改单点查询呢？只需要后两个∑\sum∑，如果只是单点修改区间查询呢？只需要前两个∑\sum∑

CDQ似乎能拜托n的限制，只不过常数有点大，至于CDQ分治怎么做嘛，学了CDQ再说。。。

Code：

#include <bits/stdc++.h>
using namespace std;

const int N = 2333;

int n, m, t[4][N][N];
int x1, y_1, x2, y2, k;

inline int lowbit(int x)
{
	return x & (- x);
}

inline void add_T(int x, int y, int k)
{
	for (int i = x; i <= n; i += lowbit(i))
		for (int j = y; j <= m; j += lowbit(j))
			t[0][i][j] += k,
			t[1][i][j] += k * x,
			t[2][i][j] += k * y,
			t[3][i][j] += k * x * y;
}

inline int ask_T(int x, int y)
{
	int res = 0;
	for (int i = x; i; i -= lowbit(i))
		for (int j = y; j; j -= lowbit(j))
			res += t[0][i][j] * (x * y + x + y + 1)
				 - t[1][i][j] * (y + 1)
				 - t[2][i][j] * (x + 1)
				 + t[3][i][j];
	return res;
}

signed main()
{
	char op[2];
	scanf("%s%d%d", op, &n, &m);
	while (scanf("%s", op) != EOF)
	{
		scanf("%d%d%d%d", &x1, &y_1, &x2, &y2);
		if (*op == 'L')
		{
			scanf("%d", &k);
		    add_T(x1, y_1, k), add_T(x1, y2 + 1, -k), add_T(x2 + 1, y_1, -k), add_T(x2 + 1, y2 + 1, k);
		}
		else
			printf("%d\n", ask_T(x2, y2) - ask_T(x2, y_1 - 1) - ask_T(x1 - 1, y2) + ask_T(x1 - 1, y_1 - 1));
	}
	
	return 0;
}

CF341D Iahub and Xors

Solution：

有了刚才那题做铺垫，这题就很水了，虽然是区间修改，但由于异或的性质，可以直接在差分数组上统计答案（一个区间的答案即为这个区间被影响到的修改的异或和）。和上一道题的区别可以形象地理解为：查询总和与数字个数有关，而异或只与一些关键点和奇偶性有关

具体实现有个细节：因为只有行列奇偶性和自己相同的修改才对自己有贡献，所以通过数组第三维和第四维，将每个点处前缀有贡献的区分了出来

总结：

最后这个细节还是挺妙的。异或与单点有关！

Code：

#include <bits/stdc++.h>
using namespace std;

const int N = 1233;

int n, m, t[N][N][2][2];
int op, x1, y1, x2, y2, k;

inline int lowbit(int x)
{
	return x & -x;
}

inline void add_T(int x, int y)
{
	int X = x & 1, Y = y & 1; 
	for (int i = x; i <= n; i += lowbit(i))
		for (int j = y; j <= n; j += lowbit(j))
			t[i][j][X][Y] ^= k;
}

inline int ask_T(int x, int y)
{
	int res = 0, X = x & 1, Y = y & 1;
	for (int i = x; i; i -= lowbit(i))
		for (int j = y; j; j -= lowbit(j))
			res ^= t[i][j][X][Y];
	return res;
}

signed main()
{
	scanf("%d%d", &n, &m);
	while (m --)
	{
		scanf("%d%d%d%d%d", &op, &x1, &y1, &x2, &y2);
		if (op & 1)
			printf("%d\n", ask_T(x1 - 1, y1 - 1) ^ ask_T(x1 - 1, y2) ^ ask_T(x2, y1 - 1) ^ ask_T(x2, y2));
		else
		{
			scanf("%d", &k);
			add_T(x1, y1), add_T(x1, y2 + 1), add_T(x2 + 1, y1), add_T(x2 + 1, y2 + 1);
		}
	}
	
	return 0;
}
