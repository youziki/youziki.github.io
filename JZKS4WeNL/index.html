题解 11.26 & 11.27 模拟赛

2020-11-27 比赛

day1 T3

非常好的一道题，主要有两点：

Solution：

Part 1 ： 观察、简化

  • 一个节点子树内没有苹果（不含这个点），这棵子树可以不存在

  • 一个节点子树内有苹果（不含这个点），如果这个点有苹果，可以忽略，因为去这个点子树一定还要经过这个点

于是树就变为了：有且只有叶子节点有苹果

问题就转化为了：将所有苹果分为两个集合，使得两个集合内边数最大值最小，ans=2×max(W1,W2)ans=2\times max(W_1,W_2)ans=2×max(W1​,W2​)

Part 2 ： dp、确定顺序

先设状态：令f[i][j][k][l]f[i][j][k][l]f[i][j][k][l]为前i个苹果，W1W1W1集合最后一个为第jjj个，W2W2W2集合最后一个为第kkk个，W1W1W1集合的大小为lll时W2W2W2集合最小的大小

现在有四个难点：如何计算集合大小，如何确定苹果顺序，如何转移，如何优化成3维状态

  1. 考虑一个确定的集合可以按照dfs序在原来的树上加入一条链，于是式子不难得出：

    W1=Dep[a1]+∑2num1Dep[ai]−Dep[lca(ai−1,ai)]W1 = Dep[a_1] + \sum_{2}^{num_1}Dep[a_i] - Dep[lca(a_{i - 1}, a_i)]W1=Dep[a1​]+∑2num1​​Dep[ai​]−Dep[lca(ai−1​,ai​)]

  2. 为了方便计算集合大小，显然应该按照dfs序将节点排序，记idiid_iidi​表示第i个苹果

  3. 考虑填表法，每次在已有状态加入一个点idi+1id_{i+1}idi+1​，枚举加入到lstlstlst所在的集合还是idiid_iidi​所在的集合，注意：转移方程里写的是实际放入点在id[]id[]id[]里的下标，计算新的大小的时候要用原点

  4. 发现W1W1W1和W2W2W2的最后一个元素中必有一个是iii，所以可以不用枚举这一维，状态里也省去，时间空间复杂度均为O(n3)O(n^3)O(n3)

因为数据范围较小（所以答案小），考虑用short压空间

Code：

/*============================================================================
  .--,       .--,   长亭外，古道边，
 ( (  \.---./  ) )  芳草碧连天。 晚风拂柳笛声残，夕阳山外山。
  '.__/o   o\__.'   天之涯，地之角，知交半零落。
     {=  ^  =}      一瓢浊酒尽余欢，今宵别梦寒。
      >  -  <
     /       \      念念不忘，必有回响
    //       \\
   //|   .   |\\    A person who is regarded as a loser isn't those
ordinarys,but the satisfieds
   "'\       /'"_.-~^`'-.
      \  _  /--'         `
    ___)( )(___
   (((__) (__)))    高山仰止,景行行止.虽不能至,心向往之。
============================================================================*/
#include <bits/stdc++.h>
using namespace std;
#define int short

const int MA = 1 << 14;
char buf[MA], *p1 = buf, *p2 = buf;
#define gc()                                                            \
    (p1 == p2 && (p2 = (p1 = buf) + fread(buf, 1, MA, stdin), p1 == p2) \
         ? EOF                                                          \
         : *p1++)
inline int read() {
    int ff = 0;
    char ch = gc();
    while (!isdigit(ch)) ch = gc();
    while (isdigit(ch)) {
        ff = (ff << 1) + (ff << 3) + (ch ^ 48);
        ch = gc();
    }
    return ff;
}
void print(int x) {
    if (x > 9) print(x / 10);
    putchar(x % 10 + '0');
}

const int N = 420;

int n, k, head[N << 2], tot;
struct E {
    int ver, nxt;
} e[N << 3];
inline void add(int u, int v) {
    e[++ tot] = {v, head[u]}, head[u] = tot;
}

int dep[N], fa[N][10], q[N];
inline void get_fa_bfs() {
    memset(dep, 0x3f, sizeof dep);
    int hh = 0, tt = 0;
    q[0] = 1, dep[0] = -1, dep[1] = 0;
    while (hh <= tt) {
        int x = q[hh ++];
        for (int i = head[x]; i; i = e[i].nxt) {
            int y = e[i].ver;
            if (dep[y] < dep[x] + 1) continue;
            dep[y] = dep[x] + 1;
            fa[y][0] = x;
            q[++ tt] = y;
            for (int k = 1; (1 << k) <= n; ++k)
                fa[y][k] = fa[fa[y][k - 1]][k - 1];
        }
    }
}
inline int lca(int U, int V) {
    if (dep[U] < dep[V]) swap(U, V);
    for (int k = 8; ~k; --k)
        if (dep[fa[U][k]] >= dep[V]) U = fa[U][k];
    if (U == V) return U;
    for (int k = 8; ~k; --k)
        if (fa[U][k] != fa[V][k]) U = fa[U][k], V = fa[V][k];
    return fa[U][0];
}

int point[N], id[N], cnt;
void get_id_dfs(int x, int F) {
    int f = point[x];
    for (int i = head[x]; i; i = e[i].nxt) {
        int y = e[i].ver;
        if (y == F) continue;
        get_id_dfs(y, x);
        point[x] += point[y];
    }
    if (f && point[x] == f) id[++ cnt] = x;
}

int f[N][N][N], L[N][N]; // “前i个叶子， 不含i的W中的最后一个叶子， 不含i的W的大小” 另一个W的最小值
inline void update(int &x, int y) {
    x = x > y ? y : x;
}
inline void dp() {
	for (int i = 0; i <= n; ++i)
		for (int j = 0; j <= n; ++j)
			for (int p = 0; p <= n; ++p)
				f[i][j][p] = 0x3f3f;
    f[1][0][0] = dep[id[1]];
    for (int i = 1; i < cnt; ++i)
        for (int lst = 0; lst < i; ++lst)
            for (int siz = 0; siz <= n; ++siz) {
            	if (lst)
                	update(f[i + 1][i][ f[i][lst][siz] ],
							siz + dep[id[i + 1]] - dep[ L[lst][i + 1] ]);
				else if (! siz)
					f[i + 1][i][ f[i][lst][siz] ] = dep[id[i + 1]];

                update(f[i + 1][lst][siz],
						f[i][lst][siz] + dep[id[i + 1]] - dep[ L[i][i + 1] ]);
			}
}

signed main() {
    n = read(), k = read();
    for (int i = 1, u, v; i < n; ++i)
        u = read(), v = read(),
        add(u, v), add(v, u);
    for (int i = 1; i <= k; ++i) ++ point[read()];
    get_fa_bfs();
    get_id_dfs(1, 0);
    for (int i = 1; i <= cnt; ++i)
    	for (int j = i + 1; j <= cnt; ++j)
    		L[i][j] = lca(id[i], id[j]);
    dp();
    int ans = 0x7f7f;
    int ff = 1;
    for (int i = 0; i < cnt; ++i) // W1最后一个是什么
        for (int j = 0; j <= n; ++j) // W1大小
        {
            ans = min(ans, int(2 * max(f[cnt][i][j], j)) );
        	if (ans == 32 && ff) {
        		ff = 0;
        		cout << i << ' ' << j << ' ' << f[cnt][i][j] << '\n';
			}
		}
	print(ans);
    return 0;
}

洛谷传送门 CF传送门

前言

模拟赛考了这题，都想的最大生成基环树森林，所以都假了，网络流可做

Solution：

Part 1 ： 为什么错

先提供一组hack数据

6 7
1 2 10000
2 3 10000
1 3 10000
4 5 10000
5 6 10000
4 6 10000
1 4 1

解释：

应该是选两个环，但最大生成基环树森林为少选一条环上的边，答案变小 （我自己写法丑，全选）

Part 2 ： 怎么改

说一下我是怎么改的，我原来的代码：

n = read(), m = read();
for (int i = 1; i <= m; ++i) {
    int a = read(), b = read(), c = read();
    e[i] = {a, b, c};
}
sort(e + 1, e + m + 1);
for (int i = 1; i <= n + m; ++i) fa[i] = i;
for (int i = 1; i <= m; ++i) {
    int a = find(e[i].a), b = find(e[i].b), c = e[i].c;
    if (a != b) ans += c, fa[min(a, b)] = max(a, b);
    else if (a <= n) ans += c, fa[a] = n + i;
}

含义：一个连通块如果已经多加了一条边，用多加的一条边作为并查集的代表元素（但这不重要，重要的是我下面怎么改）

观察到有可能a>n && b>n && a!=b，这时就会在两个环中加一条边，答案肯定变大，也不合法，因此可以判一下a和b是否全大于n（但这样还是错的）

考虑上面说到的为什么是错的，不难发现，跑最大生成树的时候最优策略为：碰到一条边，能加就加，不用同一个树内就不加了

于是判断能不能加一条边就好了，不能加当且仅当：

  • 两端为两个基环树

  • 两端为同一个基环树

用vis标记一个并查集是不是基环树即可

Code：

n = read(), m = read();
for (int i = 1; i <= m; ++i) {
    int a = read(), b = read(), c = read();
    e[i] = {a, b, c};
}
sort(e + 1, e + m + 1);
for (int i = 0; i <= n + m + 1; ++i) fa[i] = i;
for (int i = 1; i <= m; ++i) {
    int a = find(e[i].a), b = find(e[i].b), c = e[i].c;
    if (a == b) {
    	if (! vis[a])
    		vis[a] = 1, ans += c;
	}
	else {
		if (! vis[a] && ! vis[b])
			fa[a] = b, ans += c;
		else if (vis[a]) {
			if (! vis[b])
				fa[b] = a, ans += c;
		}
		else {
			fa[a] = b, ans += c;
		}
	}
}
