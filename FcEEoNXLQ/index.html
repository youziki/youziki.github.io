1 . CF1433F Zero Remainder Sum

给一个n×mn \times mn×m的矩阵，每行选择不超过一半的元素，使得最终选的所有元素和是kkk的倍数，并且和最大，求最大和 n,m,k<=70n, m, k <= 70n,m,k<=70

Solution： dpdpdp

Code：

int n, m, k;
int a[N][N], f[N][N][N][N >> 1];

int dp(int i, int j, int mod, int now)
{
    if(~ f[i][j][mod][now]) return f[i][j][mod][now];
    if(i == n + 1) return mod ? -INF : 0;
    if(j == m || now == m / 2) return f[i][j][mod][now] = dp(i + 1, 1, mod, 0);
    return f[i][j][mod][now] = max(dp(i, j + 1, mod, now), a[i][j] + dp(i, j + 1, (mod + a[i][j]) % k, now + 1));
}

signed main()
{
    memset(f, -1, sizeof f);
    n = read(), m = read(), k = read();
    for(int i = 1; i <= n; ++i)
        for(int j = 1; j <= m; ++j)
            a[i][j] = read();

    print(dp(1, 1, 0, 0));

    return 0;
}

总结： 看到这种集合元素的限制就想dpdpdp，看到不会的题，就想dpdpdp
