1 . CF1433F Zero Remainder Sum

给一个n×mn \times mn×m的矩阵，每行选择不超过一半的元素，使得最终选的所有元素和是kkk的倍数，并且和最大，求最大和 n,m,k<=70n, m, k <= 70n,m,k<=70

Solution： dpdpdp

Code：

int n, m, k;
int a[N][N], f[N][N][N][N >> 1];

int dp(int i, int j, int mod, int now)
{
    if(~ f[i][j][mod][now]) return f[i][j][mod][now];
    if(i == n + 1) return mod ? -INF : 0;
    if(j == m || now == m / 2) return f[i][j][mod][now] = dp(i + 1, 1, mod, 0);
    return f[i][j][mod][now] = max(dp(i, j + 1, mod, now), a[i][j] + dp(i, j + 1, (mod + a[i][j]) % k, now + 1));
}

signed main()
{
    memset(f, -1, sizeof f);
    n = read(), m = read(), k = read();
    for(int i = 1; i <= n; ++i)
        for(int j = 1; j <= m; ++j)
            a[i][j] = read();

    print(dp(1, 1, 0, 0));

    return 0;
}

总结： 看到这种集合元素的限制就想dpdpdp，看到不会的题，就想dpdpdp

2 . 积分（11.2 ： 没有数据）

算法标签：多重背包二进制拆分 + hash

Solution：

它到底有没有假呢？是没有的，因为模数够小，我们将所有值都判断是否能达到，因为数据范围够小且数据够水，这样是对的

Code：

#include <bits/stdc++.h>
using namespace std;
#define LL long long

const int mod = 100007;
const int base = 7;
const int N = 233333;

int n, k, tot, val[N];
bitset <mod> f;

signed main()
{
	ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
	
	cin >> n >> k;
	int a, b, c, d, e;
	for (int i = 1; i <= n; ++i)
	{
		cin >> a >> b >> c >> d >> e;
		a = 1LL * base * base * base * base * a % mod;
		b = 1LL * base * base * base * b % mod;
		c = 1LL * base * base * c % mod;
		d = base * d % mod;
		e = e % mod;
		int va = (1LL * a + b + c + d + e) % mod, x = k;
		for (int j = 1; j <= x;  x -= j, j <<= 1)
			val[++tot] = va * j % mod;
		if(x) val[++tot] = x;
	}
	
	cin >> a >> b >> c >> d >> e;
	a = 1LL * base * base % mod * base * base * a % mod;
	b = 1LL * base * base % mod * base * b % mod;
	c = 1LL * base * base * c % mod;
	d = base * d % mod;
	e = e % mod;
	int ans = (1LL * a + b + c + d + e) % mod;
	
	f[0] = 1;
	for(int i = 1; i <= tot; ++i)
		f |= f << val[i];
	puts(f[ans] ? "SEY" : "ON");
	
	return 0;
}

P5098 [USACO04OPEN]Cave Cows 3

Solution：

Code：

#include <cstdio>
#include <algorithm>

const int inf=1<<30;

int main() {
    int n,a=-inf,b=inf,c=-inf,d=inf;
    for(scanf("%d",&n);n--;) {
        int x,y;
        scanf("%d%d",&x,&y);
        a=std::max(a,x+y);
        b=std::min(b,x+y);
        c=std::max(c,x-y);
        d=std::min(d,x-y);
    }
    printf("%d\n",std::max(a-b,c-d));
    return 0;
}
