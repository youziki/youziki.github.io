题解 CF1433C 【Dominant Piranha】

2020-10-30

传送门

这题做的时候真傻了，想了半天各种复杂做法，然后发现其实这么简单，导致D和E切得晚了...

引理：选择最大数

证明：

  • 如果序列全为最大数，无解

  • 否则序列定有一段最大数且这一段的边界旁边会有一个小一点的数，于是可以让边界的最大数吃它，然后这个数变成整个序列唯一的最大数，显然是答案

代码模拟（找到这一位置）即可，记得开long long

Code：

#include<bits/stdc++.h>
using namespace std;
#define int long long
const int N = 3e5 + 7;
int a[N];
signed main()
{
	int t;
	scanf("%lld", &t);
	while(t --)
	{
		int n, mx = 0;
		scanf("%lld", &n);
		for(int i = 1; i <= n; ++i)
		{
			scanf("%lld", &a[i]);
			mx = max(mx, a[i]);
		}
		int f = 1, lst = a[1];
		for(int i = 2; i <= n && f; ++i)
			if(a[i] != lst) f = 0;
		if(f)
		{
			puts("-1");
			continue;
		}
		for(int i = 1; i <= n && ! f; ++i)
			if(a[i] == mx && i != 1 && a[i - 1] < a[i])
			{
				printf("%lld\n", i);
				f = 1;
			}
			else if(a[i] == mx && a[i + 1] != mx && i != n){
				printf("%lld\n",i);
				f = 1;
			}

		for(int i = 1; i <= n; ++i) a[i] = 0;
	}
	
	return 0;
}
