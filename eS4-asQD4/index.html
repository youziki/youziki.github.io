P3385 【模板】负环

一定要判入队次数，即在判!vis[y]! vis[y]!vis[y]之后，不能判松弛次数

Code：

#include <bits/stdc++.h>
using namespace std;

const int N = 23333;

int n, m, head[N], tot, q[N], cnt[N], dis[N];
bool vis[N];

struct E
{
	int ver, nxt, w;
} e[N << 2];

inline void add(int u, int v, int w)
{
	e[++ tot] = {v, head[u], w}, head[u] = tot;
}

inline bool spfa()
{
	memset(cnt, 0, sizeof cnt);
	memset(dis, 0x3f, sizeof dis);
	memset(vis, 0, sizeof vis);
	
	int hh = 0, tt = 1;
	q[0] = 1, dis[1] = 0, vis[1] = 1;
	
	while(hh != tt)
	{
		int x = q[hh ++];
		if(hh == N - 2) hh = 0;
		vis[x] = 0;
		
		for(int i = head[x]; i; i = e[i].nxt)
		{
			int y = e[i].ver, z = e[i].w;
			if(dis[y] > dis[x] + z)
			{
				dis[y] = dis[x] + z;
				if(! vis[y])
				{
					vis[y] = 1;
					if(++ cnt[y] > n) return 1;
					q[tt ++] = y;
					if(tt == N - 2) tt = 0;
				}
			}
		}
	}
	
	return 0;
}

signed main()
{
	int t;
	scanf("%d", &t);
	while(t --)
	{
		memset(head, 0, sizeof head);
		
		scanf("%d%d", &n, &m);
		for(int i = 1; i <= m; ++i)
		{
			int u, v, w;
			scanf("%d%d%d", &u, &v, &w);
			add(u, v, w);
			if(w >= 0) add(v, u, w);
		}
		
		puts(spfa() ? "YES" : "NO");
	}
	
	return 0;
}

P1938 [USACO09NOV]Job Hunt S

spfaspfaspfa跑最长路的时候将原边权减去，点权加上即可

Code：

#include <bits/stdc++.h>
using namespace std;

const int N = 2333;

int d, p, c, f, s, head[N], tot, q[N], dis[N];
bool vis[N];

struct E
{
	int ver, nxt, w;
} e[N];

inline void add(int u, int v, int w)
{
	e[++ tot] = {v, head[u], w}, head[u] = tot;
}

inline void spfa()
{
	memset(dis, 0xcf, sizeof dis);
	memset(vis, 0, sizeof vis);
	
	int hh = 0, tt = 1;
	q[0] = s, dis[s] = d, vis[s] = 1;
	
	while(hh != tt)
	{
		int x = q[hh ++];
		if(hh == N - 2) hh = 0;
		vis[x] = 0;
		
		for(int i = head[x]; i; i = e[i].nxt)
		{
			int y = e[i].ver, z = e[i].w;
			if(dis[y] < dis[x] - z + d)
			{
				dis[y] = dis[x] - z + d;
				if(! vis[y])
				{
					vis[y] = 1;
					q[tt ++] = y;
					if(tt == N - 2) tt = 0;
				}
			}
		}
	}
}

signed main()
{
	scanf("%d%d%d%d%d", &d, &p, &c, &f, &s);
	for(int i = 1; i <= p; ++i)
	{
		int u, v;
		scanf("%d%d", &u, &v);
		add(u, v, 0);
	}
	for(int i = 1; i <= f; ++i)
	{
		int u, v, w;
		scanf("%d%d%d", &u, &v, &w);
		add(u, v, w);
	}
	
	spfa();
	
	int ans = 0;
	for(int i = 1; i <= c; ++i) ans = max(ans, dis[i]);
	printf("%d", ans);
	
	return 0;
}

P6175 无向图的最小环问题

Floyd原理：前k-1轮中得到了前k-1个数之间的最短路（只经过前k-1个数），将一个环中最大的点删去，剩下的一定是两端两个点之间的最短路，且不经过大于等于k的点，所以在每一轮k的时候枚举两个点i,j即可

Code：

#include <bits/stdc++.h>
using namespace std;
#define int long long

const int N = 105;

int n, m, a[N][N], mp[N][N];

signed main()
{
	scanf("%lld%lld", &n, &m);
	
	for(int i = 1; i <= n; ++i)
		for(int j = 1; j <= n; ++j)
			mp[i][j] = a[i][j] = 2e10;
	for(int i = 1; i <= n; ++i) mp[i][i] = a[i][i] = 0;
	
	for(int i = 1; i <= m; ++i)
	{
		int u, v, w;
		scanf("%lld%lld%lld", &u, &v, &w);
		mp[u][v] = mp[v][u] = a[u][v] = a[v][u] = min(a[u][v], w);
	}
	
	int ans = 2e10;
	for(int k = 1; k <= n; ++k)
	{
		for(int i = 1; i < k; ++i)
			for(int j = i + 1; j < k; ++j)
				ans = min(ans, a[i][j] + mp[i][k] + mp[k][j]);
		
		for(int i = 1; i <= n; ++i)
			for(int j = 1; j <= n; ++j)
				a[i][j] = min(a[i][j], a[i][k] + a[k][j]);
	}

	if(ans == 2e10) puts("No solution.");
	else printf("%lld", ans);
	
	return 0;
}

板子们：

SPFA：O(m)O(m)O(m)

定义变量

int n, m, S, T, dis[N], head[N], tot, q[N]; // 一个距离数组，一个队列，一个标记数组
bool vis[N];

核心函数

inline void spfa()
{
    memset(dis, 0x3f, sizeof dis); // 最短路

    int hh = 0, tt = 1;
    dis[S] = 0, q[0] = S, vis[S] = 1; // 初始化

    while(hh != tt) // 注意这里的写法、hh和tt的初值、以及后面的hh ++和tt ++
    {
        int x = q[hh ++];
        if(hh == N - 2) hh = 0; // 循环数组
        vis[x] = 0;

        for(int i = head[x]; i; i = e[i].nxt)
        {
            int y = e[i].ver, z = e[i].w;
            if(dis[y] > dis[x] + z)
            {
                dis[y] = dis[x] + z;
                if(! vis[y]) // 如果不在队中，进队
                {
                    vis[y] = 1;
                    q[tt ++] = y;
                    if(tt == N - 2) tt = 0; // 别忘了循环数组
                }
            }
        }
    }
}

朴素dij：O(n2)O(n^2)O(n2)

定义变量

int a[N][N], dis[N]; // 距离数组、标记数组
bool vis[N];

核心函数

inline void dij()
{
    dis[S] = 1;
    for (int i = 1; i <= n; ++i)
    { // 每次取出一个点
        int t = -1;
        for (int j = 1; j <= n; ++j) // 没有被取出的dis最小的点
            if (!vis[j] && (t == -1 || dis[t] < dis[j])) t = j;
        vis[t] = 1; // 标记
        for (int j = 1; j <= n; ++j) dis[j] = max(dis[j], dis[t] + a[t][j]); // 更新
    }
}

bfs（边权为0/1）：O(n)O(n)O(n)

定义变量

int n, m, a[N][N], dis[N], q[N]; // 一个距离数组，一个队列

核心函数

inline void bfs()
{
    memset(dis, 0x3f, sizeof dis);

    int hh = 0, tt = 0;
    q[0] = 1, dis[1] = 0; // 初始化

    while(hh <= tt)
    {
        int x = q[hh ++];
        for(int i = 1; i <= n; ++i)
            if(a[x][i] && dis[i] > dis[x] + 1)
            {
                dis[i] = dis[x] + 1; // 直接更新，不需要vis，每个点第一次入队就是最短路
                // 所以所有点都入队结束，最短路更新完毕
                q[++ tt] = i;
            }
    }
}

P1455 搭配购买

Solution：

会做题，也要学会用简单方法做简单题，这题并查集完全可以搞定，都没必要连边！

Code：

#include <bits/stdc++.h>
using namespace std;

const int N = 23333;

int n, m, V, fa[N], c[N], d[N], f[N];

int find(int x)
{
	return x == fa[x] ? x : fa[x] = find(fa[x]);
}

signed main()
{
	scanf("%d%d%d", &n, &m, &V);
	for (int i = 1; i <= n; ++i) scanf("%d%d", c + i, d + i);
	
	for (int i = 1; i <= n; ++i) fa[i] = i;
	
	for (int i = 1; i <= m; ++i)
	{
		int u, v;
		scanf("%d%d", &u, &v);
		u = find(u), v = find(v);
		if (u != v)
		{
			fa[u] = v;
			c[v] += c[u], d[v] += d[u];
		}
	}
	
	for (int i = 1; i <= n; ++i)
		if (find(i) == i)
			for (int j = V; j >= c[i]; --j)
				f[j] = max(f[j], f[j - c[i]] + d[i]);
	
	printf("%d", f[V]);
	
	return 0;
}

P1892 [BOI2003]团伙

Solution：

扩展域并查集板子，比“食物链”还简单，扩展域并查集合并的部分具体谁的父亲指向谁很迷，说明一下。

看以下样例：

3
2
E 1 2
E 2 3

图：

模拟一遍程序可以发现，在连完绿边之后，连蓝边之前findfindfind反集时已经将反集用根节点（正集）表示，所以连出的边能完全表示正集的关系（可能说的有点迷）

Code：

#include <bits/stdc++.h>
using namespace std;
int n,m,fa[1000000];
int find(int x)
{
	return x==fa[x]?x:fa[x]=find(fa[x]);
}
signed main()
{
	cin>>n>>m;
	for(int i=1;i<=n*2;++i)fa[i]=i;
	for(int i=1;i<=m;++i)
	{
		int u,v;
		char op;
		cin>>op>>u>>v;
		int uu=find(u+n),vv=find(v+n);
		u=find(u),v=find(v);
		if(op=='F')fa[u]=v;
		else fa[uu]=v,fa[vv]=u;
	}
	int ans=0;
	for(int i=1;i<=n;++i)if(find(i)==i)++ans;
	cout<<ans;
	
	return 0;
}
