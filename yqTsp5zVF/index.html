题解 CF1406B 【Maximum Product】

2020-10-30 3 min read # 题解

本菜鸡的第一场CF（div2）

本题不算难，但是前几次心急，导致写了删删了写......浪费了好长时间，思路清晰就不难了 （别老想着模拟）

solution

记录选了几个正数，然后去个最大值即可 ansx代表选了x个正数

时间复杂度考试时候没算 （现在也没算） 肯定是能过的，不会超过O(NlogN)吧......

坑点：

  • 一定要开long long！

  • 需要赋值负无穷大，无穷大取1e19 取1<<30会错aaa!

  • 看咱数组范围！当然不能memset清空

code:

const int N=1e5+7;
LL inf=1e19;
int t,n,cnt1,cnt0,a[N],b[N];
il bool cmp(int x,int y){return x>y;}
signed main(){
	read(t);
	while(t--){
		int x;
		LL ans0=-inf,ans1=-inf,ans2=-inf,ans3=-inf,ans4=-inf,ans5=-inf,ans=-inf;
		cnt1=cnt0=0;
		read(n);
		REP(i,1,n){
			read(x);
			if(x>0)a[++cnt1]=x;else b[++cnt0]=x;
		}
		sort(a+1,a+cnt1+1,cmp); sort(b+1,b+cnt0+1);
		if(cnt1>=5)ans5=(LL)a[1]*a[2]*a[3]*a[4]*a[5];
		if(cnt1>=4&&cnt0>=1)ans4=(LL)b[cnt0]*a[cnt1]*a[cnt1-1]*a[cnt1-2]*a[cnt1-3];
		if(cnt1>=3&&cnt0>=2)ans3=(LL)b[1]*b[2]*a[1]*a[2]*a[3];
		if(cnt1>=2&&cnt0>=3)ans2=(LL)b[cnt0]*b[cnt0-1]*b[cnt0-2]*a[cnt1]*a[cnt1-1];
		if(cnt1>=1&&cnt0>=4)ans1=(LL)b[1]*b[2]*b[3]*b[4]*a[1];
		if(cnt0>=5)ans0=(LL)b[cnt0]*b[cnt0-1]*b[cnt0-2]*b[cnt0-3]*b[cnt0-4];
		REP(i,1,cnt1)a[i]=0; REP(i,1,cnt0)b[i]=0;
		ans=max(ans,ans0);
		ans=max(ans,ans1);
		ans=max(ans,ans2);
		ans=max(ans,ans3);
		ans=max(ans,ans4);
		ans=max(ans,ans5);
		printf("%lld\n",ans);
	}
  	return 0;
}
