题解 11.17 模拟赛

2020-11-17 比赛

上午模拟赛太难，就先不写了

T5 平面图

Solution：

考场上想到思路，估计错复杂度，把O(2mmn)O(2^mmn)O(2mmn)估成了O(2mn2)O(2^mn^2)O(2mn2)，导致自暴自弃写了prim/kk

实际上kruskal就能跑满分，只不过用到MST的性质，挺好想的，循环内跑MST的时候只遍历最小生成树内的边

Code：

#include <bits/stdc++.h>
using namespace std;
const int N = 2005, M = 15;
int n, m, tot, cnt, x[N], y[N], k[N], c[N], son[M][N], fa[N], sum, ed[N];
int find(int x) {return x == fa[x] ? x : fa[x] = find(fa[x]);}
struct Edge {
	int a, b, c;
	bool operator < (const Edge &A) const {return c < A.c;}
} e[N * N];
inline int work() {
	int res = 0;
	for (int i = 1; i <= cnt; ++i) {
		int a = find(e[ed[i]].a), b = find(e[ed[i]].b);
		if (a != b) res += e[ed[i]].c, fa[a] = b;
	}
	return res;
}
signed main() {
	scanf("%d%d", &n, &m); for (int i = 1; i <= n; ++i) scanf("%d%d", x + i, y + i);
	for (int i = 1, u, v; i <= m; ++i) {
		scanf("%d%d", k + i, c + i);
		for (int j = 1; j <= k[i]; ++j) scanf("%d", &son[i][j]);
	}
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j <= n; ++j) if (i != j)
			e[++ tot] = {i, j, (x[i] - x[j]) * (x[i] - x[j]) + (y[i] - y[j]) * (y[i] - y[j])};
	sort(e + 1, e + tot + 1);
	for (int i = 1; i <= n; ++i) fa[i] = i;
	for (int i = 1; i <= tot; ++i) {
		int a = find(e[i].a), b = find(e[i].b), c = e[i].c;
		if (a != b) fa[a] = b, sum += c, ed[++ cnt] = i;
	}
	int ans = sum, res = 0;
	for (int i = 1; i < (1 << m); ++i) {
		res = 0;
		for (int j = 1; j <= n; ++j) fa[j] = j;
		for (int j = 1; j <= m; ++j) if (i & (1 << (j - 1))) {
			int F = find(son[j][1]);
			for (int l = 2; l <= k[j]; ++l) if (find(son[j][l]) != F) fa[find(son[j][l])] = F;
			res += c[j];
		}
		ans = min(ans, work() + res);
	}
	printf("%d", ans);
	return 0;
}
