题解 CF1406A 【Subset Mex】

2020-10-30 1 min read # 题解

solution

第一个集合选择从0到原集合价值-1的元素

第二个集合剩下的元素

ans=第一个集合价值+第二个集合价值

一个集合的价值可以O(N)算出

code：（自认为简洁易懂）

const int N=110;
int t,n,a[N],b[N];
signed main(){
	read(t);
	while(t--){
		int now=0,ans=0;
		memset(a,0,sizeof(a)); memset(b,0,sizeof(b));
		read(n); REP(i,1,n){read(a[i]);++b[a[i]];}
		REP(i,0,101)if(b[i])--b[i];else{now=i;break;}
		ans=now;
		REP(i,0,101)if(!b[i]){now=i;break;}
		printf("%d\n",ans+now);
	}
	return 0;
}
