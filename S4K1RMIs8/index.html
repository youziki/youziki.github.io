题解 P5336 【[THUSC2016]成绩单】

2020-10-30 题解

讲题时现理解出的一些东西，一本通里并没有说明白

Solution：

设状态为f[x][y][l][r]f[x][y][l][r]f[x][y][l][r]表示把区间[x,y][x,y][x,y]中数值不在[l,r][l,r][l,r]内的点取完的最小代价（显然后两维要离散化），“不在...内”是因为一段区间的代价可以通过一次操作加上A+B×(r−l)2A + B \times (r - l) ^ 2A+B×(r−l)2轻松得到

具体怎么转移，先看样例

10
3 1
7 10 9 10 6 7 10 7 1 2

ans = 15

正确的取法：
10 9 10
10
1 6 7 7
1 2

可以发现，如果可以取任意多次（A相对B足够小但不为0），那肯定是按照值域取，先将最大的取完（连续的1次取完），这时一些次大的数有可能会连起来，就可以让取的次数减少，再这样取次大的、一直到取完

但实际情况并不如此乐观，所以我们按照刚刚的思想，重新定义“取完”为：

“将区间[x,y][x,y][x,y]取完，指这个区间内每次取一段值域（这些数的位置并不一定连续），直至取完”

于是就可以得到l = 0 &&r = 0时的转移

f[x][y][0][0]=min(A+B×(ma−mi)2,∑i=1tot∑j=itotmin(f[x][y][i][j]+A+B×(j−i)2))f[x][y][0][0] = min(A + B \times (ma - mi)^2,\sum_{i = 1}^{tot} \sum_{j = i} ^ {tot} min(f[x][y][i][j] + A + B \times (j - i)^2))f[x][y][0][0]=min(A+B×(ma−mi)2,∑i=1tot​∑j=itot​min(f[x][y][i][j]+A+B×(j−i)2))

tottottot是值域

这个式子中需要用到：f[x][y][i][j]f[x][y][i][j]f[x][y][i][j]，现在我们来推这个情况

如果区间内只有一个数，如果不在[i,j][i,j][i,j]内，代价显然为0，否则，它需要被取，因为f[x][y][i][j]f[x][y][i][j]f[x][y][i][j]和f[x][y][0][0]f[x][y][0][0]f[x][y][0][0]是同时

如果区间所有数都不在[l,r][l,r][l,r]之内，不用管这种情况

反之，要满足[l,r][l,r][l,r]的限制，一定是由一段一段的“取区间”操作组成的，也就是一些f[xi][yi][0][0]f[xi][yi][0][0]f[xi][yi][0][0]组成的，我们要划分成一些区间使得这些数加起来最小

枚举中间断点，划分为左右两个区间，有三种情况：

  • 左边继续划分，右边直接取完

  • 左边直接取完，右边继续划分

  • 左右都继续划分

于是就有了转移方程：

f[x][y][l][r]=∑k=xy−1min(f[x][k][l][r]+f[k+1][y][0][0],f[x][k][0][0]+f[k+1][y][l][r],f[x][k][l][r]+f[k+1][y][l][r])f[x][y][l][r] = \sum_{k = x}^{y - 1} min(f[x][k][l][r] + f[k + 1][y][0][0],f[x][k][0][0] + f[k + 1][y][l][r],f[x][k][l][r] + f[k + 1][y][l][r])f[x][y][l][r]=∑k=xy−1​min(f[x][k][l][r]+f[k+1][y][0][0],f[x][k][0][0]+f[k+1][y][l][r],f[x][k][l][r]+f[k+1][y][l][r])

记搜实现，因为一本通就用的记搜这样好写

Code：

#include<bits/stdc++.h>
using namespace std;
const int N = 55, INF = 1e8;

int n, A, B, tot, f[N][N][N][N], a[N], tmp[N], b[N];
bool vis[N][N][N][N];

int dp(int x, int y, int l, int r)
{
	if(vis[x][y][l][r]) return f[x][y][l][r];
	vis[x][y][l][r] = 1;

	if(!l && !r)
	{
		int ma = -INF, mi = INF;
		for(int i = x;i <= y; ++i)
		{
			ma = max(ma, a[i]);
			mi = min(mi, a[i]);
		}
		f[x][y][l][r] = A + B * (ma - mi) * (ma - mi);

		for(int i = 1;i <= tot; ++i)
			for(int j = i;j <= tot; ++j)
				f[x][y][l][r] = min(f[x][y][l][r], dp(x, y, i, j) + A + B * (tmp[j] - tmp[i]) * (tmp[j] - tmp[i]));

		return f[x][y][l][r];
	}
	else
	{
		if(x == y)
		{
			if(l <= b[x] && b[x] <= r) return f[x][y][l][r] = 0;
			return f[x][y][l][r] = A;
		}

		f[x][y][l][r] = INF;
		for(int k = x;k < y; ++k)
		{
			f[x][y][l][r] = min(f[x][y][l][r], dp(x, k, l, r) + dp(k + 1, y, l, r));
			f[x][y][l][r] = min(f[x][y][l][r], dp(x, k, 0, 0) + dp(k + 1, y, l, r));
			f[x][y][l][r] = min(f[x][y][l][r], dp(x, k, l, r) + dp(k + 1, y, 0, 0));
		}
		return f[x][y][l][r];
	}
}

signed main()
{
	scanf("%d %d %d", &n, &A, &B);
	for(int i = 1;i <= n; ++i)
	{
		scanf("%d", &a[i]);
		tmp[i] = a[i];
	}

	sort(tmp + 1, tmp + n + 1);
	tot = unique(tmp + 1, tmp + n + 1) - (tmp + 1);
	for(int i = 1;i <= n; ++i)
		b[i] = lower_bound(tmp + 1, tmp + tot + 1, a[i]) - tmp;

	printf("%d", dp(1, n, 0, 0));
	return 0;
}
