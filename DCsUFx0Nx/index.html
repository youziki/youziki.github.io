T2 直播获奖

难度：黄/绿

Solution：

  1. 树状数组+二分 O(nlog2n)O(nlog^2 n)O(nlog2n)

  2. 正解：对顶堆 O(nlogn)O(nlogn)O(nlogn)

    我这个sb居然还想的是对顶栈！

    这可以说是一个对顶堆板子题， 好好学着：

    • 右边一个小根堆，为了让分界线处元素为堆顶元素，左边一个大根堆，为了让分界线左边一个元素为堆顶元素

    • 右边堆的堆顶时刻保持是分界线处的元素（根据题目不同有时为分界线左边有时为分界线右边），本题为分界线右边

    • 图中的序列是从小到大排好序的（实际上不排），可以发现，题中实际上就是求右边符合条件的部分最靠左的元素，即右边堆的堆顶

    • 新插入一个元素x，如果x<右边堆的堆顶元素或右边堆为空，就将x加入右边堆，否则加入左边堆（在“排序”好的数组中加入一个元素是多么的简单！）

    • 关键来了：如何维护左右两堆

      算出新的右边堆应该的siz（橙色部分），如果右边堆元素大于这个，弹堆顶到左边堆（相当于分界线在新排序好的数组中移动），同理，如果不足，就从左边堆弹堆顶到右边堆

    • 时间复杂度证明：每次插入元素是O(nlogn)O(nlogn)O(nlogn)，分界线移动的总次数不会超过O(n)O(n)O(n)，所以也是O(nlogn)O(nlogn)O(nlogn)，总时间复杂度O(nlogn)O(nlogn)O(nlogn)

多么美妙的算法！

Code：

#include <bits/stdc++.h>
using namespace std;

signed main()
{
    int n, w;
    scanf("%d%d", &n, &w);
    
    priority_queue<int> left;
    priority_queue<int, vector<int>, greater<int> > right;
    
    for (int i = 1; i <= n; ++i)
    {
        int x;
        scanf("%d", &x);
        if (right.empty() || x >= right.top()) right.push(x);
        else left.push(x);
        
        int k = max(1, int(i / 100.0 * w));
        while (right.size() > k) left.push(right.top()), right.pop();
        while (right.size() < k) right.push(left.top()), left.pop();
        
        printf("%d ", right.top());
    }
    
    return 0;
}

T4 方格取数

难度：蓝

Solution：

dpdpdp吼题啊！

想半个duo小时可以想到用前缀和做，从左往右dpdpdp，代码有点恶心，一遍70，时间复杂度O(n2m)O(n^2 m)O(n2m)

实在想不出正解O(nm)O(nm)O(nm)做法：

每到一个点的状态可以是从左边直接过来的或是从上面下来的，或是从下面上来的，所以直接按位置dpdpdp，从左往右dpdpdp，从上往下统计一遍最优解，从下往上再更新一遍，保证每个位置的三种情况都考虑过了

注意：第二遍dpdpdp的时候要用s变量辅助记录到上一个位置（下方）的最优解，因为s可以保证是从下或左转移而来的，而如果用f[i+1][j]f[i + 1][j]f[i+1][j]进行更新，其表示的含义有可能是从上面下来的最优解，再上去，与题意不符

Code：

#include <bits/stdc++.h>
using namespace std;
#define LL long long

const int N = 1005;

int n, m, a[N][N];
LL f[N][N];

signed main()
{
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n; ++i)
        for (int j = 1; j <= m; ++j)
            scanf("%d", &a[i][j]);
    
    memset(f, 0xcf, sizeof f);
    f[1][0] = 0;
    for (int j = 1; j <= m; ++j)
    {
        for (int i = 1; i <= n; ++i)
            f[i][j] = max(f[i][j - 1], f[i - 1][j]) + a[i][j];
        LL s = -1e18;
        for (int i = n; i; --i)
        {
            s = max(s, f[i][j - 1]) + a[i][j];
            f[i][j] = max(f[i][j], s);
        }
    }
    
    printf("%lld", f[n][m]);
    
    return 0;
}

T3 函数调用

难度：蓝

有点绕

Solution：

最终的答案一定可以表示为ai×X+Ya_i \times X + Yai​×X+Y，XXX都是一样的，直接看操作二的个数就行了，问题就变成了如何找每个位置的YYY，YYY可以看成是若干次不同的加操作组合成的，因为有操作2，所以每个加操作有不同的系数

不难看出，函数的调用关系构成了一个DAG，于是可以想到用拓扑排序来做。具体一点，每个操作序列分成只有操作一和操作二后，在后面的操作一的执行次数为1，在后面的操作2的影响就是让前面的操作（不只是操作一）的次数×vi\times v_i×vi​，于是就可以先拓扑一遍，找出所有操作执行的次数。

从前面的分析中可以发现，应该从后执行的操作往前拓扑，因为后执行的操作不会受到影响，所以要记得将操作三的操作顺序反过来

在统计答案的时候，用suf[]suf[]suf[]记录每个操作执行后乘的总系数，tag[]tag[]tag[]表示每个操作执行的次数.

注意，在循环到一个操作的时候，要tag[per[i]] = (tag[per[i]] + suf[i + 1]) % mod;而不是乘上suf[i+1]suf[i + 1]suf[i+1]，因为tag[]tag[]tag[]现在的值是前面操作中这个操作的执行次数，加上这次操作中的执行次数，而这次操作中的执行次数就为1×suf[i+1]1 \times suf[i + 1]1×suf[i+1]

Code：

#include <bits/stdc++.h>
using namespace std;

const int mod = 998244353;
const int maxn = 1e5 + 5;

int per[maxn], suf[maxn];
int n, m, q;
int a[maxn], op[maxn], p[maxn], v[maxn], c[maxn];
int deg[maxn], tag[maxn], ans[maxn], mul[maxn];
vector<int> E[maxn];

void dfs(int x)
{
    if (mul[x]) return;
    mul[x] = op[x] == 2 ? v[x] : 1;
    for (int i = 0; i < E[x].size(); ++i)
    {
    	int v = E[x][i];
        dfs(v);
        mul[x] = 1ll * mul[x] * mul[v] % mod;
    }
}

void work(int x)
{
    if (op[x] == 1) (ans[p[x]] += 1ll * v[x] * tag[x] % mod) %= mod;
    int tmp = tag[x];
    for (int i = 0; i < E[x].size(); ++i)
    {
    	int v = E[x][i];
        (tag[v] += tmp) %= mod;
        if (--deg[v] == 0) work(v);
        tmp = 1ll * tmp * mul[v] % mod;
    }
}

signed main()
{
    scanf("%d", &n);
    for (int i = 1; i <= n; ++i) scanf("%d", a + i);
    scanf("%d", &m);
    for (int i = 1; i <= m; ++i)
    {
        scanf("%d", op + i);
        if (op[i] == 1) scanf("%d%d", p + i, v + i);
        else if (op[i] == 2) scanf("%d", v + i);
        else if (op[i] == 3)
        {
            scanf("%d", c + i);
            while (c[i]--)
            {
                int x;
                scanf("%d", &x);
                deg[x]++;
                E[i].push_back(x);
            }
        }
    }
    for (int i = 1; i <= m; i++) reverse(E[i].begin(), E[i].end());
    for (int i = 1; i <= m; ++i)
        if (!deg[i])
        {
            E[m + 1].push_back(i);
            deg[i]++;
        }
    dfs(m + 1);
    
    scanf("%d", &q);
    suf[q + 1] = 1;
    for (int i = 1; i <= q; ++i)
    {
        scanf("%d", per + i);
        suf[i] = mul[per[i]];
    }
    for (int i = q; i; --i)
    {
        tag[per[i]] = (tag[per[i]] + suf[i + 1]) % mod;
        suf[i] = 1ll * suf[i + 1] * suf[i] % mod;
    }
    work(m + 1);
    for (int i = 1; i <= n; ++i) printf("%d ", (1ll * a[i] * suf[1] % mod + ans[i]) % mod);

    return 0;
}

T4 贪吃蛇

难度：黑

它的70分做法大概也就紫题难度（最多），O(Tnlogn)O(Tnlogn)O(Tnlogn)，正解对排序优化后不用setsetset或priorityqueuepriority_queuepriorityq​ueue维护，O(Tn)O(Tn)O(Tn)，但是正解证明繁琐难懂（没看懂），而且不必深究偏、难的部分，所以只写了前者，核心思想已经有了

Solution：

考虑不管在哪一个位置停下，它的状态是唯一确定的，也就是说，我们可以强制让最强的吃最弱的，知道只剩一个蛇，最后停止一定是在中间达到的某一个状态停止。

做法：

  1. 先模拟出所有状态，可以得到每一个蛇的死亡时间，因为要动态维护最小值和最大值，用set维护。（用当前剩多少个蛇表示一个时间点）

  2. 记一个lastalllivelast_all_livelasta​lll​ive表示最晚在这个时间点，所有蛇可以活，初值为1。倒推，如果到一个时间点，这个时间点最大的蛇的死亡时间大于lastalllivelast_all_livelasta​lll​ive说明假如这个蛇不停止，让时间变小，这个蛇会在后面被吃掉（因为这个蛇的死亡时间肯定大于当前时间，而且后面吃掉这个蛇后的时间仍不小于lastalllivelast_all_livelasta​lll​ive，即剩下的其他蛇都可以存活，所以这个蛇肯定会被吃掉），所以一旦大于，这个蛇就会停止，那么前面哪些蛇都能存活，将lstalllivelst_all_livelsta​lll​ive赋为当前这个时间

  3. 最后，将所有死亡时间小于lastalllivelast_all_livelasta​lll​ive的蛇加入答案即可

至于优化，可以发现就在模拟处优化排序即可，类似“蚯蚓”，用两个双端队列一个维护原蛇的大小，另一个维护吃完之后的蛇的大小，目标是做到“O(1)O(1)O(1)插入队尾、删除/查询最大最小”，中间有些证明和处理很难，就不会了

Code：（加上快读开O2之后，洛谷能AC）

#include <bits/stdc++.h>
using namespace std;

#define pii pair<int, int>
#define m_p make_pair

const int N = 1233333;

int a[N], win[N], death[N];
set<pii> s;

signed main()
{
	int T = read(), n, I = 0;
	while (I++ < T)
	{
		if (I == 1)
		{
			n = read();
			for (int i = 1; i <= n; ++i) a[i] = read();
		}
		else
		{
			int k = read();
			for (int i = 1; i <= k; ++i)
			{
				int x = read(), y = read();
				a[x] = y;
			}
		}
		
		s.clear();
		for (int i = 1; i <= n; ++i) s.insert(s.end(), m_p(a[i], i));
		for (int i = n; i >= 2; --i)
		{
			auto it = (-- s.end());
			win[i] = it -> second;
			int sval = it -> first;
			s.erase(it);
			death[s.begin() -> second] = i;
			int wval = s.begin() -> first;
			s.erase(s.begin());
			s.insert(m_p(sval - wval, win[i]));
		}
		death[win[2]] = 1;
		
		int last_all_live = 1, ans = 0;
		for (int i = 3; i <= n; ++i)
			if (death[win[i]] > last_all_live)
				last_all_live = i;
		for (int i = 1; i <= n; ++i) ans += death[i] <= last_all_live;

		printf("%d\n", ans);
	}
	
	return 0;
}
