#define int long long

const int N = 100005;
const int mx = 1e9 + 7;

int rt[N * 300], tot, sum[N * 300], ls[N * 300], rs[N * 300];

int update(int root, int l, int r, int k)
{
    int dir = ++ tot;
    sum[dir] = sum[root], ls[dir] = ls[root], rs[dir] = rs[root];
    if(l == r)
    {
        sum[dir] += k;
        return dir;
    }
    int mid = (l + r) >> 1;
    if(k <= mid)
        ls[dir] = update(ls[dir], l, mid, k);
    else
        rs[dir] = update(rs[dir], mid + 1, r, k);
    sum[dir] = sum[ls[dir]] + sum[rs[dir]];
    return dir;
}

int query(int p, int q, int l, int r, int L, int R)
{
    if(R < l || r < L) return 0;
    if(L <= l && r <= R) return sum[q] - sum[p];
    int mid = (l + r) >> 1;
    return query(ls[p], ls[q], l, mid, L, R) + query(rs[p], rs[q], mid + 1, r, L, R);
}

inline int calc(int l, int r)
{
    int maxa = 0, suma = 0, t; // maxa表示之前用到的最大的a，防止同一个值重复加入，suma表示之前的前缀和
    while(t = query(rt[l - 1], rt[r], 1, mx, maxa + 1, suma + 1))
        maxa = suma + 1, suma += t;
    return suma + 1;
}

signed main()
{
    int n = read(), m = read();
    for(int i = 1; i <= n; ++i)
        rt[i] = update(rt[i - 1], 1, mx, read());
    while(m --)
    {
        int a = read(), b = read();
        cout << calc(a, b) << endl;
    }
    
    return 0;
}

显然将时间离散化，考虑dpdpdp

设f[i]f[i]f[i]表示iii时间段在iii位置最早的时间（每个时间段只一个限制中对应一个位置）

因为可以来回走，似乎不好转移，考虑最短路，实质上就是在状态之间跑最短路，堆中按ttt排（ttt相当于最短路中的距离，时间段和位置都是一个点）

#define pii pair<int, int>
#define L first
#define R second
#define m_p make_pair

const int N = 200010;
const int INF = 0x3f3f3f3f;

vector <pii> v[N], cur;
int n, m, id[N], f[N];

struct P
{
	int id, x, t;

	inline bool operator > (const P &A) const
	{
		return t > A.t;
	}
};
priority_queue < P, vector <P>, greater <P> > q;

inline void init()
{
	n = read(), m = read();
	for(int i = 1; i <= m; ++i)
		v[read()].push_back( {read(), read()} );
	v[0].push_back( {0, INF} );
	v[n + 1].push_back( {0, INF} );

	id[1] = 1;
	for(int i = 1; i <= n; ++i)
	// 把v[i]变成了第i个位置的空档（门是打开的）
	// 预处理出id[]表示i位置第一个空档编号
	{
		sort(v[i].begin(), v[i].end());
		cur.clear();
		int r = -1;
		for(int j = 0; j < v[i].size(); ++j)
		{
			auto p = v[i][j];
			if(p.L > r + 1) cur.push_back( {r + 1, p.L - 1} );
			r = max(r, p.R);
		}
		cur.push_back( {r + 1, INF} );
		v[i] = cur;
		id[i + 1] = id[i] + v[i].size();
	}
}

inline void calc(P p, int x) // 转移
{
	int r = v[p.x][p.id - id[p.x]].R; // 注意vector下标从0开始，这里是找到从这个位置，最晚可以在什么时候走到x
	int i = lower_bound(v[x].begin(), v[x].end(), m_p(p.t + 1, 0) ) - v[x].begin() - 1; // 找出可以更新的状态

	if(v[x][i].R >= p.t + 1 && f[id[x] + i] > p.t + 1) // 如果可以更新，而且更新更优，更新，并加入新状态
	{
		f[id[x] + i] = p.t + 1;
		q.push( (P){id[x] + i, x, p.t + 1});
	}
	
	for(++ i; i < v[x].size() && v[x][i].L <= r + 1; ++i)
		if(f[id[x] + i] > v[x][i].L) // 从i往后任意时刻到x都可以，就把每一段都更新为开头
		{
			f[id[x] + i] = v[x][i].L;
			q.push( (P){id[x] + i, x, v[x][i].L});
		}
}

inline void dp()
{
	memset(f, 0x3f, sizeof f);
	f[0] = 0;
	q.push( (P){0, 0, 0} ); // 起点入队（初始化）

	while(q.size())
	{
		auto p = q.top(); q.pop();
		if(p.t > f[p.id]) continue;
		if(p.x > 0) calc(p, p.x - 1); // 后退
		if(p.x < n + 1) calc(p, p.x + 1); // 前进
        // 最终是要走到n + 1的
	}
}

signed main()
{
	init();
	dp();
	print( f[id[n + 1]] );

	return 0;
}

const int N = 1000007;

int n, m, head[N], tot, dep[N];
int eu[N], lg[N], st[N], dfn, minn[N][25];

struct E
{
    int ver, nxt;
} e[N];

inline void add(int u, int v)
{
    e[++ tot] = {v, head[u]}, head[u] = tot;
}

void dfs(int x, int fa)
{
    dep[x] = dep[fa] + 1;
    eu[++ dfn] = x;
    st[x] = dfn;
    for(int i = head[x]; i; i = e[i].nxt)
    {
        int y = e[i].ver;
        if(y == fa) continue;
        dfs(y, x);
        eu[++ dfn] = x;
    }
}

inline int mi(int x, int y)
{
    return dep[x] < dep[y] ? x : y;
}

inline void lca_pre()
{
    for(int i = 1; i <= dfn; ++i) minn[i][0] = eu[i];
    int mx = n << 1;
    for(int i = 1; i <= lg[mx]; ++i)
        for(int j = 1; j + (1 << i) - 1 <= mx; ++j)
            minn[j][i] = mi( minn[j][i - 1], minn[j + (1 << (i - 1))][i - 1] );
}

inline int lca(int x, int y)
{
    if(st[x] > st[y]) swap(x, y);
    int l = st[x], r = st[y], k = lg[r - l + 1];
    return mi( minn[l][k], minn[r - (1 << k) + 1][k] );
}

inline int dis(int x, int y)
{
    return dep[x] + dep[y] - (dep[lca(x, y)] << 1);
}

struct P
{
    int t, pos;
} p[N], now[N], pp[N];

int cntp, vis[N], mii[N], tag, cntnow, block;

inline void spfa()
{
    queue <P> q;
    q.push(p[1]);
    int l = 2;
    while(q.size())
    {
        auto x = q.front(); q.pop();
        for(int i = head[x.pos]; i; i = e[i].nxt)
        {
            int y = e[i].ver;
            if(! vis[y])
            {
                vis[y] = 1;
                mii[y] = min(mii[y], mii[x.pos] + 1);
                q.push( (P){mii[y], y} );
                if(mii[y] == p[l].t)
                {
                    vis[p[l].pos] = 1;
                    q.push(p[l ++]);
                }
            }
        }
    }
}

inline void RE()
{
    memset(vis, 0, sizeof vis);
    memset(mii, 0x7f, sizeof(mii));

    for(int i = 1; i <= cntnow; ++i) p[++ cntp] = now[i];
    for(int i = 1; i <= cntp; ++i)
        mii[p[i].pos] = min(mii[p[i].pos], p[i].t);
    
    spfa();
}

inline void query(int x, int times)
{
    if(! tag && times >= mii[x])
    {
        puts("wrxcsd");
        return;
    }
    for(int i = 1; i <= cntnow; ++i)
        if(dis(x, now[i].pos) <= times - now[i].t)
        {
            puts("wrxcsd");
            return;
        }
    puts("orzFsYo");
}

inline void init()
{
    memset(mii, 0x7f, sizeof mii);

    lg[0] = -1;
    for(int i = 1; i <= N - 6; ++i)
        lg[i] = lg[i >> 1] + 1;

    n = read(), m = read();
    p[0].t = 1e9;
    for(int i = 1; i < n; ++i)
    {
        int u = read(), v = read();
        add(u, v), add(v, u);
    }
    dfs(1, 0);
    lca_pre();
    block = sqrt(m) * 2;
}

inline void work()
{
    for(int i = 1; i <= m; ++i)
    {
        if(i % block == 0)
        {
            RE();
            cntnow = tag = 0;
        }
        int op = read(), x = read();
        if(op == 1)
