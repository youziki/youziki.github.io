做题历程（应该会给写挂这题的您一些参考）：

看完题，心想：连边最短路（搜索太烂了，就没想搜索）

第一次写完之后：样例输出正无穷

原因：部分装置和普通点之间边没有连

第二次改完：同上

原因：装置之间反复横跳

改进的方法：不管必须走装置的限制（可以拿38分）

如何让它又满足必须走装置又可以从装置往外走呢？：拆点！

写过网络流的大佬应该很熟悉拆点

“就是将一个点拆成一个入点和一个出点，对于入点，只有入度，对于出点，只有出度，最后入点再连向出点”

不同的题拆法不一样，所以以上方法辅助理解

知道拆点之后，不难想到这题即把每套装置的两端点拆开，连边法则如下（有点绕）：

用A、B来表示同一个装置两个端点

  • A端点出点：负责连出（连向除B端点入点外的所有该连的点点）

  • A端点入点：只连向B端点出点（所有不相干的点连向入点）

  • B端点出点：负责连出（连向除A端点入点外的所有该连的点点）

  • B端点入点：只连向A端点出点（所有不相干的点连向入点）

理一下思路：
可能存在一条合法路径如下：

1.个普通点（草地、起点、终点皆为普通点）连向A点入点，

2.走装置”限制，走到B点出点，

3.点出点可以走到其他的普通点

（需要明确的是：
反复经过一个装置没有意义）

这题已经理论上被解决了！

刚才只是介绍了连边的原理 （您大概还是不能一遍切掉）

成品连边代码就这么多啦，还是有一些需要说明的地方

  • 首先就是为什么不都在输入的for循环里处理了

    1. 注意到在连普通到普通的时候，是一定可以连双向边的，

      所以我们只需要与它之前的点连双向边即可（结合代码理解，这就是为什么只枚举了上、左格子）

    2. 对于装置连装置，只与他们两个端点有关，因为记录了map，变得很好处理，不会混淆，所以在for里解决

    3. 而对于装置互相连普通点的情况，

      因为装置连向普通点与普通点连向装置是不同的含义（连向的点情况比较复杂），

      所以不能简单用后面连接双向边代替前面需要连的边

      选择在图都输入完之后，对于一个装置端点，直接找周围的普通点，一次连到位

    4. 对于装置连向另一个装置端点这么复杂的情况，当然也是同理放在外面处理

  • 其次就是将一个装置端点拆点时的细节，

    我为了避免麻烦，直接给出点加上了总点数

    （可以发现，就算你加上1e8，应该也不会有问题，不用考虑空间，节点编号不重要）

连边结束，这题就解决了，跑dij最短路就行了（SPFA应该也可以）

#include<bits/stdc++.h>
#define re register
#define int long long
#define il inline
using namespace std;
template<typename T>il void read(T &ff){
    T rr=1;ff=0;re char ch=getchar();
    while(!isdigit(ch)){if(ch=='-')rr=-1;ch=getchar();}
    while(isdigit(ch)){ff=(ff<<1)+(ff<<3)+(ch^48);ch=getchar();}
    ff*=rr;
}
const int N=1e6+7;
int n,m,num,tot,head[N],S,T,dis[N],vis[N],flag[N];
priority_queue<pair<int,int> > q;
map<char,int> MAP;
struct E{
    int ver,nxt,w;
}e[N<<1];
il void add(re int u,re int v,re int w){
    e[++tot]=(E){v,head[u],w};
    head[u]=tot;
}
char ch[310][310];
il void Dijkstra(re int s){
    memset(dis,0x3f,sizeof(dis));
	memset(vis,0,sizeof(vis));
	dis[s]=0;
	q.push(make_pair(0,s));
	while(q.size()){
		int x=q.top().second;q.pop();
		if(vis[x])continue;
		vis[x]=1;
		for(int i=head[x];i;i=e[i].nxt){
			int y=e[i].ver,z=e[i].w;
			if(dis[y]>dis[x]+z){
				dis[y]=dis[x]+z;
				q.push(make_pair(-dis[y],y));
			}
		}
	} 
}
il int calc(re int i,re int j){//二维坐标转节点编号
    return (i-1)*m+j;
}
il bool check(re int i,re int j){
    return ch[i][j]=='.'||ch[i][j]=='@'||ch[i][j]=='=';
}
il bool check2(re int i,re int j){
    return ch[i][j]>='A'&&ch[i][j]<='Z';
}
signed main(){
    read(n),read(m);
    num=n*m;
    for(re int i=1;i<=n;++i){
        scanf("%s",ch[i]+1);
        for(re int j=1;j<=m;++j){
            //标记起点、终点
            if(ch[i][j]=='=')T=calc(i,j);
            if(ch[i][j]=='@')S=calc(i,j);
            if(check(i,j)){
                //普通跟普通（全连）（只需连与它前面点连双向边）
                if(check(i-1,j))
                    add(calc(i,j),calc(i-1,j),1),add(calc(i-1,j),calc(i,j),1);
                if(check(i,j-1))
                    add(calc(i,j),calc(i,j-1),1),add(calc(i,j-1),calc(i,j),1);
            }
            if(check2(i,j)){//特殊跟特殊（+num为出点，入点连出点）
                if(MAP.find(ch[i][j])!=MAP.end())
                    add(calc(i,j),MAP[ch[i][j]]+num,0),add(MAP[ch[i][j]],calc(i,j)+num,0);
                else MAP[ch[i][j]]=calc(i,j);
            }
        }
    }
    for(re int i=1;i<=n;++i)
        for(re int j=1;j<=m;++j)
            if(check2(i,j)){
                //特殊跟普通（出点连即可）、普通连特殊（连入点（不+num））
                if(check(i-1,j))
                    add(calc(i,j)+num,calc(i-1,j),1),add(calc(i-1,j),calc(i,j),1);
                if(check(i,j-1))
                    add(calc(i,j)+num,calc(i,j-1),1),add(calc(i,j-1),calc(i,j),1);
                if(check(i+1,j))
                    add(calc(i,j)+num,calc(i+1,j),1),add(calc(i+1,j),calc(i,j),1);
                if(check(i,j+1))
                    add(calc(i,j)+num,calc(i,j+1),1),add(calc(i,j+1),calc(i,j),1);
                
                //特殊跟另一个特殊（想不到吧）（出点连向入点） 
				if(check2(i-1,j))
                    add(calc(i,j)+num,calc(i-1,j),1),add(calc(i-1,j)+num,calc(i,j),1);
				if(check2(i,j-1))
                    add(calc(i,j)+num,calc(i,j-1),1),add(calc(i,j-1)+num,calc(i,j),1);
				if(check2(i+1,j))
                    add(calc(i,j)+num,calc(i+1,j),1),add(calc(i+1,j)+num,calc(i,j),1);
				if(check2(i,j+1))
                    add(calc(i,j)+num,calc(i,j+1),1),add(calc(i,j+1)+num,calc(i,j),1);
        }
    Dijkstra(S);
    printf("%d",dis[T]);
    return 0;
}
