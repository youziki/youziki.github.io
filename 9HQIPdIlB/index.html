Codeforces Round #668 (Div. 1)

A. Balanced Bitstring

题面：

给一个01串，中间有一些问号（可以为0也可以为1），是否可以变成“每一个长度为k的子串01个数相等”的串

Solution：

枚举前k个字符，每一个往后每次+k+k+k，遇到不合法的就停止，遇到必须赋值的就立刻赋值（为了后面遇到不合法的可以判出来），不合法一定是前k个0或1个数大于k2\frac k 22k​，输出即可

B. Tree Tag

Solution：

博弈论，结论很简单，怎么都想不出来，证明也不是很难

只有三种情况 Alice 赢

dis(a,b)≤dadis(a,b) ≤ dadis(a,b)≤da
2×da≥db2×da≥db2×da≥db 此时只要一步步逼近即可
树的直径 ≤2×da≤2×da≤2×da 此时只要占据直径中点即可

以下证明剩下情况必定可以（设 b 为根，特殊链指的是长度大于 2×da2×da2×da 的链）

  • 若特殊链某一端点到 b 路径上（不含 b） 的任意节点不是 a 的祖先

    那么直接对着这一端点跳然后特殊链上反复横跳即可

  • 若特殊链在 a 子树内

    则 a 往子树内最深处 d 走至少能走 da+1 步，加上 a,b 距离至少为 da+1，故 b 到 d 的链一定为特殊链且为 1. 情况，成立

  • 若特殊链一端点到 b 路径不经过 a 且除 b 外还有一个点 c 为 a 祖先

    若 dadis(a,c)>dadadis(a,c)>dadadis(a,c)>da ，则显然存在 1. 情况链，否则直接向 c 沿链（非 a 方向）走显然 Alice 追不上，走到底之后就可以在这条特殊链上反复横跳了

Solution

Code1：
#include <bits/stdc++.h>
using namespace std;
#define int long long

const int N = 300005;
char ch[N];

signed main() {
    int t;
    cin >> t;
    while (t--) {
        int n, k, f = 0;
        cin >> n >> k >> ch + 1;
        for (int i = 1; i <= k && !f; ++i)
            for (int j = i + k; j <= n; j += k)
                if (ch[i] != '?' && ch[j] != '?' && ch[i] != ch[j])
                    f = 1;
                else if (ch[i] == '?' && ch[j] != '?')
                    ch[i] = ch[j];
        if (f) {
            puts("NO");
            continue;
        }
        int c0 = 0, c1 = 0;
        for (int i = 1; i <= k; ++i) c0 += ch[i] == '0', c1 += ch[i] == '1';
        if (c0 > k / 2 || c1 > k / 2)
            puts("NO");
        else
            puts("YES");
    }
    return 0;
}

inline int read() {
    int ff = 0;
    char ch = gc();
    while (!isdigit(ch)) ch = gc();
    while (isdigit(ch)) {
        ff = (ff << 1) + (ff << 3) + (ch ^ 48);
        ch = gc();
    }

}

const int N = 100005;
int n, m, posa, posb, disa, disb, head[N], tot, dep[N], id, maxdep;
struct E {
    int ver, nxt;
} e[N << 1];
inline void add(int u, int v) { e[++tot] = {v, head[u]}, head[u] = tot; }
void dfs1(int x, int F) {
    dep[x] = dep[F] + 1;
    maxdep = max(maxdep, dep[x]);
    for (int i = head[x]; i; i = e[i].nxt) {
        int y = e[i].ver;
        if (y == F) continue;
        dfs1(y, x);
    }
}
signed main() {
    int t = read();
    while (t--) {
        for (int i = 1; i <= n; ++i) head[i] = 0;
        tot = maxdep = 0;
        n = read(), posa = read(), posb = read(), disa = read(), disb = read();
        for (int i = 1, u, v; i < n; ++i)
            u = read(), v = read(), add(u, v), add(v, u);
        if (disa * 2 >= disb) {
            puts("Alice");
            continue;
        }
        dep[0] = -1;
        dfs1(posb, 0);
        if (dep[posa] <= disa) {
            puts("Alice");
            continue;
        }
        for (int i = 1; i <= n; ++i) {
            if (maxdep == dep[i]) id = i;
            dep[i] = 0;
        }
        dfs1(id, 0);
        puts(maxdep <= disa * 2 ? "Alice" : "Bob");
    }
    return 0;
}
