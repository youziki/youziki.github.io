T2 序列问题

题面：

求所有子区间最大值最小值的乘积之和 n<=105n <= 10^5n<=105

Solution：

考场刚了半天才想到单调栈，又写了之后才发现不能维护一段相同最大值内的不同最小值，昂神立刻就想到了维护最大值最小值的乘积，所以就发现不用树状数组或线段树维护最大最小值是不行的（区间赋值，区间求和），单调栈写法没调出来，全WA，过样例（复杂度O(nlogn)O(nlogn)O(nlogn)速度还挺快，最慢的点500ms）

正解：分治O(nlogn)O(nlogn)O(nlogn)

但是怎样才能保证复杂度呢？对左半部分每一个位置i，它的最大最小值在右区间内找到一个相同的值，因为两边的最大最小值都具有单调性，所以可以用双指针O(n)O(n)O(n)对左区间每个i找到这样的位置，下面说说怎样处理一个区间：

答案分三部分统计：

  • 对于浅蓝色的区间，最大最小值直接为i处的最大最小值，直接乘即可

  • 对于深蓝色的区间，最大值固定，记录最小值前缀和相乘即可（有可能此处为：min<mi,max=mxmin<mi,max=mxmin<mi,max=mx，但只需要令最小值固定，乘最大值前缀和即可）

  • 对于紫色的区间，最大值最小值为区间内的，直接记录最大最小值乘积前缀和即可

#include <bits/stdc++.h>
using namespace std;
#define int long long
#define LL long long
const int N = 500005, P = 1e9 + 7;
int n, a[N];
int stk[2][N], top[2], wid[2][N]; // 减、增
LL t[2][2][N]; // 第一维：mx, mi 第二维：aj, aj*j
LL ans, sum;
inline void add_T(LL t[], LL x, LL y) {
	for (; x < N; x += x & -x) t[x] = (t[x] + y) % P ;
}
inline LL ask_T(LL t[], LL x) {
	LL res = 0;
	for (; x; x -= x & -x) res = (res + t[x]) % P;
	return res;
}
inline void add_1(int l, int r, LL y, int k) { // 区间加（差分、维护两个树状数组）
	add_T(t[k][0], l, y), add_T(t[k][1], l, y * l);
	add_T(t[k][0], r + 1, -y), add_T(t[k][1], r + 1, -y * (r + 1));
}
inline LL ask_1(int l, int r, int k) { // 区间查询（式子），记得前缀和相减
	LL ans1 = (1LL * (r + 1) * ask_T(t[k][0], r) % P - ask_T(t[k][1], r) + P) % P;
	LL ans2 = (1LL * l * ask_T(t[k][0], l - 1) % P - ask_T(t[k][1], l - 1) + P) % P;
	return (ans1 - ans2 + P) % P;
}
signed main() {
	scanf("%lld", &n);
	for (int i = 1; i <= n; ++i) {
		scanf("%lld", a + i);
		if (! a[i]) {
			ans = 0;
			while (top[0]) wid[0][top[0]] = stk[0][top[0]] = 0, -- top[0];
			while (top[1]) wid[1][top[1]] = stk[1][top[1]] = 0, -- top[1];
			memset(t, 0, sizeof t);
			continue;
		}
		ans = (ans + 1LL * a[i] * a[i] % P) % P;
		if (a[i] < a[ stk[0][top[0]] ]) stk[0][++ top[0]] = i, wid[0][top[0]] = 1, add_1(i, i, a[i], 0);
		else {
			int width = 0;
			while (top[0] && a[ stk[0][top[0]] ] <= a[i]) {
				LL s = ask_1(stk[0][top[0]] - wid[0][top[0]] + 1, stk[0][top[0]], 1);
				ans = (ans + 1LL * (a[i] - a[ stk[0][top[0]] ]) * s % P) % P;
				add_1(stk[0][top[0]] - wid[0][top[0]] + 1, stk[0][top[0]], - a[ stk[0][top[0]] ] + a[i], 0);
				width += wid[0][top[0]], -- top[0];
			}
			stk[0][++ top[0]] = i, wid[0][top[0]] = width + 1;
		}
		if (a[i] > a[ stk[1][top[1]] ]) stk[1][++ top[1]] = i, wid[1][top[1]] = 1, add_1(i, i, a[i], 1);
		else {
			int width = 0;
			while (top[1] && a[ stk[1][top[1]] ] >= a[i]) {
				LL s = ask_1(stk[1][top[1]] - wid[1][top[1]] + 1, stk[1][top[1]], 0);
				ans = (ans + 1LL * (a[i] - a[ stk[1][top[1]] ]) * s % P + P) % P;
				add_1(stk[1][top[1]] - wid[1][top[1]] + 1, stk[1][top[1]], - a[ stk[1][top[1]] ] + a[i], 1);
				width += wid[1][top[1]], -- top[1];
			}
			stk[1][++ top[1]] = i, wid[1][top[1]] = width + 1;
		}
		sum = (sum + ans) % P;
	}
	printf("%lld", sum);
	return 0;
}

#include <bits/stdc++.h>
using namespace std;
#define int long long
const int N = 500005, P = 1000000007;
int n, ans, a[N], mi[N], mx[N], sum[2][N], sum2[N];
inline int add(int x, int y) {return x + y >= P ? x + y - P : x + y;}
inline int del(int x, int y) {return x > y ? x - y : x - y + P;}
void solve(int l, int r) {
	if (l > r) return;
	if (l == r) return ans = add(ans, a[l] * a[l] % P), void();
	int mid = (l + r) >> 1;
	mi[mid] = P, mx[mid] = -P;
	for (int i = mid + 1; i <= r; ++i)
		mi[i] = min(mi[i - 1], a[i]), mx[i] = max(mx[i - 1], a[i]),
		sum[0][i] = add(sum[0][i - 1],  mi[i]), sum[1][i] = add(sum[1][i - 1], mx[i]),
		sum2[i] = add(sum2[i - 1], mi[i] * mx[i] % P);
	int mn_p = mid + 1, ma_p = mid + 1, mn = P, ma = -P;
	for (int i = mid; i >= l; --i) {
		mn = min(mn, a[i]); while (mi[mn_p] > mn && mn_p <= r) ++ mn_p;
		ma = max(ma, a[i]); while (mx[ma_p] < ma && ma_p <= r) ++ ma_p;
		ans = add(ans, mn_p < ma_p ? add(mn * ma % P * (mn_p - mid - 1) % P, add(ma * del(sum[0][ma_p - 1], sum[0][mn_p - 1]) % P, del(sum2[r], sum2[ma_p - 1])))
								   : add(mn * ma % P * (ma_p - mid - 1) % P, add(mn * del(sum[1][mn_p - 1], sum[1][ma_p - 1]) % P, del(sum2[r], sum2[mn_p - 1]))) );
	}
	for (int i = mid + 1; i <= r; ++i) mi[i] = mx[i] = sum[0][i] = sum[1][i] = sum2[i] = 0;
	solve(l, mid), solve(mid + 1, r);
}
signed main() {
	scanf("%lld", &n); for (int i = 1; i <= n; ++i) scanf("%lld", a + i);
	solve(1, n);
	printf("%lld", ans);
	return 0;
}

#include <bits/stdc++.h>
using namespace std;
const int N = 21, P = 1e9 + 7;
int n, m, k, mx, f[1 << N][N], ans;
bitset<N> nxt[N];
signed main() {
	scanf("%d%d%d", &n, &m, &k); mx = (1 << n) - 1;
	for (int i = 1, u, v; i <= m; ++i) scanf("%d%d", &u, &v), nxt[u].set(v - 1);
	for (int i = 1; i <= n; ++i) f[1 << (i - 1)][0] = 1;
	for (int s = 0; s <= mx; ++s)
		for (int j = 1; j <= n; ++j) if (! ((s >> (j - 1)) & 1) )
			for (int p = 0; p <= k; ++p) {
				bitset<N> cur(s);
				cur &= nxt[j];
				int p2 = p + cur.count();
				if (p2 <= k) {
					int &x = f[s | (1 << (j - 1))][p2];
					x = (x + f[s][p]) % P;
				}
			}
	for (int i = 0; i <= k; ++i) ans = (ans + f[mx][i]) % P;
	printf("%d", ans);
	return 0;
}
