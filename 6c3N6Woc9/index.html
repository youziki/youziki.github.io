P1972 [SDOI2009]HH的项链：基础莫队

询问静态区间内不同颜色个数，离线

笔记：

块大小：n2m\sqrt\frac {n^2}{m}mn2​​

结构体：左右端点，询问编号（id）和左端点所在块编号（bl）

排序：第一关键字：左端点所在块，第二关键字：右端点（奇块从小到大，偶块从大到小）

Code：

定义变量、重载排序：

int n, m, t, a[N], sum[N], ans[N], cnt;
struct Q {
	int l, r, id, bl;
	inline bool operator < (const Q &A) const {
		return bl != A.bl ? l < A.l : (bl & 1 ? r < A.r : r > A.r);
	}
} q[N];

init：

scanf("%d", &n);
for (int i = 1; i <= n; ++i) scanf("%d", a + i);
scanf("%d", &m);
t = sqrt(n * n / m);
for (int i = 1, u, v; i <= m; ++i) {
    scanf("%d%d", &u, &v);
    q[i] = {u, v, i, (u - 1) / t + 1};
}

莫队核心部分：四个while循环（不要写成if）、注意l和r的初值

sort(q + 1, q + m + 1);
int l = 1, r = 0;
for (int i = 1; i <= m; ++i)
{
    int ll = q[i].l, rr = q[i].r;
    while (l < ll) cnt -= ! -- sum[a[l ++]];
    while (l > ll) cnt += ! sum[a[-- l]] ++;
    while (r < rr) cnt += ! sum[a[++ r]] ++;
    while (r > rr) cnt -= ! -- sum[a[r --]];
    ans[q[i].id] = cnt;
}
for (int i = 1; i <= m; ++i) printf("%d\n", ans[i]);

P1903 [国家集训队]数颜色 / 维护队列：带修莫队

询问区间内不同颜色数，每次单点修改，离线

笔记：

块大小：(nm)13(nm) ^ \frac 1 3(nm)31​

结构体：

  • 询问：左右端点，前面有多少个修改（表示“时间”）（pre），第几个询问（id），左右端点所在块编号

  • 修改：修改位置，修改前值，修改后值（注意：进行这个修改的时候，前面的修改一定都已经做完了，所以这个修改前值应该是该位置上一个修改之后的值，而不是一开始的值）

排序：第一关键字：左端点所在块，第二关键字：右端点所在块，第三关键字：时间（升序）

原理：

多了时间一维，像左右端点一样，从前一个区间的时间开始移动指针到当前时间即可，时间复杂度：O(n53)O(n ^ \frac 5 3)O(n35​)（默认n==m）

写起来有些麻烦，主要差别在变量和与处理上

一些细节：

输入的时候，输入一个修改操作，要及时修改为正确值，是为了正确初始化修改的结构体（pre）

输入完之后，要从后往前将修改还原

Code：

定义变量

int n, m, t, a[N], ans[N], sum[M], cnt, totc, totq;
struct C {
	int pos, pre, suf;
} cg[N];
struct Q {
	int l, r, pre, id, bll, blr;
	bool operator < (const Q &A) const {
		return bll != A.bll ? bll < A.bll : (blr != A.blr ? blr < A.blr : pre < A.pre);
	}
} q[N];

init

scanf("%d%d", &n, &m);
for (int i = 1; i <= n; ++i) scanf("%d", a + i);
for (int i = 1, u, v; i <= m; ++i) {
	char ch[2];
	scanf("%s%d%d", ch, &u, &v);
	if (*ch == 'R') ++ totc, cg[totc] = {u, a[u], v}, a[u] = v;
	else ++ totq, q[totq] = {u, v, totc, totq};
}
t = max(1, (int)pow(n * totq, 3 / 1.0));
for (int i = 1; i <= totq; ++i) q[i].bll = (q[i].l - 1) / t + 1, q[i].blr = (q[i].r - 1) / t + 1;
for (int i = totc; i; --i) a[cg[i].pos] = cg[i].pre;

核心函数

sort(q + 1, q + totq + 1);
int l = 1, r = 0, t = 0;
for (int i = 1; i <= totq; ++i) {
    // 这里和普通莫队一样
	int ll = q[i].l, rr = q[i].r, tt = q[i].pre;
	while (l < ll) cnt -= ! -- sum[a[l ++]];
	while (l > ll) cnt += ! sum[a[-- l]] ++;
	while (r < rr) cnt += ! sum[a[++ r]] ++;
	while (r > rr) cnt -= ! -- sum[a[r --]];
    // 时间往后推和时间回溯，注意posi不要改了
	while (t < tt) {
		int posi = cg[++ t].pos;
		if (posi >= l && posi <= r) cnt -= ! -- sum[a[posi]];
		a[posi] = cg[t].suf;
		if (posi >= l && posi <= r) cnt += ! sum[a[posi]] ++;
	}
	while (t > tt) {
		int posi = cg[t].pos;
		if (posi >= l && posi <= r) cnt -= ! -- sum[a[posi]];
		a[posi] = cg[t --].pre;
		if (posi >= l && posi <= r) cnt += ! sum[a[posi]] ++;
	}
	ans[q[i].id] = cnt;
}

AT1219 歴史の研究：回滚莫队

询问静态区间最大值，离线

Code：
Code1：
#include <bits/stdc++.h>
using namespace std;
#define int long long
const int N = 100005, M = 2000005;
int n, m, k, t, xo[N], sum[M], cnt, ans[N];
struct Q {
	int l, r, id, bl;
	inline bool operator < (const Q &A) const {return bl != A.bl ? bl < A.bl : (bl & 1 ? r < A.r : r > A.r);}
} q[N];
signed main() {
	scanf("%lld%lld%lld", &n, &m, &k);
	for (int i = 1; i <= n; ++i) scanf("%lld", xo + i + 1), xo[i + 1] ^= xo[i];
	t = max((int)1, (int)sqrt(n * n / m));
	for (int i = 1, u, v; i <= m; ++i) {
		scanf("%lld%lld", &u, &v);
		q[i] = {u, v, i, (u - 1) / t + 1};
	}
	sort(q + 1, q + m + 1);
	int l = 1, r = 0;
	for (int i = 1; i <= m; ++i) {
		int ll = q[i].l, rr = q[i].r + 1;
		while (l < ll) -- sum[xo[l]], cnt -= sum[xo[l ++] ^ k];
		while (l > ll) cnt += sum[xo[-- l] ^ k], ++ sum[xo[l]];
		while (r < rr) cnt += sum[xo[++ r] ^ k], ++ sum[xo[r]];
		while (r > rr) -- sum[xo[r]], cnt -= sum[xo[r --] ^ k];
		ans[q[i].id] = cnt;
	}
	for (int i = 1; i <= m; ++i) printf("%lld\n", ans[i]);
	return 0;
}
