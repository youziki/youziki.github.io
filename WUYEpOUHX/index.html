T2 淘淘蓝蓝与保温箱

也挺水的，纯暴力就能过，就是转移和初始化的时候常规费了点心思

Code：

#include <bits/stdc++.h>
using namespace std;

const int N = 85;

int n, q, k, suma, sumb;
int a[N], b[N], f[N][N][N * 20], g[N];

int main()
{
	freopen("box.in", "r", stdin);
	freopen("box.out", "w", stdout);
	
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i) scanf("%d", a + i), suma += a[i];
	for (int i = 1; i <= n; ++i) scanf("%d", b + i), sumb += b[i];
	
	// dp
	memset(f, 0xcf, sizeof f);
	f[0][0][0] = 0;
	for (int i = 1; i <= n; ++i)
	{
		for (int j = 0; j <= i; ++j)
			for (int k = 0; k <= sumb; ++k)
				f[i][j][k] = f[i - 1][j][k];
		for (int j = 1; j <= i; ++j)
			for (int k = b[i]; k <= sumb; ++k)
				f[i][j][k] = max(f[i][j][k], f[i - 1][j - 1][k - b[i]] + a[i]);
	}
	
	scanf("%d", &q);
	memset(g, 0x3f, sizeof g);
	while (q --)
	{
		scanf("%d", &k);
		if (g[k] < 0x3f3f3f3f)
		{
			printf("%d\n", g[k]);
			continue;
		}
		int ans = -1;
		for (int i = suma; i <= sumb; ++i)
			ans = max(ans, f[n][k][i]);
		printf("%d\n", g[k] = (ans <= -1 ? -1 : suma - ans));
	}
	
	fclose(stdin);
	fclose(stdout);
	return 0;
}
/*
选不超过k个
总体积大于等于总水量
总价值最大
f[i][j][k]前i个(80)，选了j个(80)，体积为k(80*20)，最大价值(2)
*/

#include <bits/stdc++.h>
using namespace std;
#define pii pair<int, int>

const int N = 2333;
const int INF = 1e8;

char ch[N][N];
int n, m, dis[N][N], sx, sy, px = INF, py;
pii q[N];
int facx[8] = {-1, 1, 0, 0, -1, -1, 1, 1}, facy[8] = {0, 0, -1, 1, -1, 1, -1, 1};

inline void init()
{
	scanf("%d%d", &n, &m);
	for (int i = 1; i <= n; ++i) scanf("%s", ch[i] + 1);
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j <= m; ++j)
			if (ch[i][j] == '*') sx = i, sy = j;
			else if (ch[i][j] == 'X' && px > i) px = i, py = j;
	
	for (int i = 1; i < px; ++i) ch[i][py] = 'X';
}

inline bool check(int x, int y)
{
	if (x < 1 || y < 1) return 0;
	if (x > n || y > m) return 0;
	if (ch[x][y] == 'X') return 0;
	return 1;
}

#define f dis
queue<pair<int, int> > q;
void spfa()
{
    memset(f, 0x3f, sizeof(f));
    f[sx][sy] = 0;
    q.push(make_pair(sx, sy));
    while (q.size())
	{
        int x = q.front().first, y = q.front().second;
        q.pop();
        for (int i = 0; i < 8; ++i)
		{
            int tx = x + facx[i], ty = y + facy[i];
            if (tx >= 1 && tx <= n && ty >= 1 && ty <= m && f[x][y] + 1 < f[tx][ty] && ch[tx][ty] != 'X')
			{
                f[tx][ty] = f[x][y] + 1;
                q.push(make_pair(tx, ty));
            }
        }
    }
}

inline void calc()
{
	int ans = INF;
	for (int i = 1; i < px; ++i)
	{
		int t1 = min( dis[i][py + 1], min( dis[i + 1][py + 1], dis[i - 1][py + 1] ) );
		int t2 = min( dis[i][py - 1], min( dis[i + 1][py - 1], dis[i - 1][py - 1] ) );
		ans = min(ans, t1 + t2 + 2);
	}

}
