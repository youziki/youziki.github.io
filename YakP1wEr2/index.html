CF1045D Interstellar battle

真的是很好的题！又思路又套路，还清晰简单 （还让人思考的时候觉得很难很难）

如果dpdpdp的话是树形dpdpdp，还是期望dpdpdp，还是动态dpdpdp。。。

按照组合数，自己想到了考虑分成了几块？不行，每一块没法表示和讨论

有一个套路：对于一个块，用一个顶点表示

这句话我一开始也没有看懂，它在本题中的应用是：

考虑将一个连通块的价值归到父节点上，一个节点有价值当且仅当它的父节点被删除且它没有被删除，将所有节点的价值加起来就是答案

对于一个节点的价值，我们根据上面的定义就很容易得到：

∑yp[x]×(1−p[y])\sum_{y} p[x] \times (1 - p[y])∑y​p[x]×(1−p[y])

由这个式子，我们可以得到一个TrickTrickTrick：

只维护∑y×(1−p[y])\sum_{y} \times (1 - p[y])∑y​×(1−p[y])，这样修改p[x]p[x]p[x]的复杂度就变为O(1)O(1)O(1)的啦

每次修改对答案的贡献很好算：ans+=sum[x]×(p1−p[x])−(p1−p[x])×fa[x]ans += sum[x] \times (p1 - p[x]) - (p1 - p[x]) \times fa[x]ans+=sum[x]×(p1−p[x])−(p1−p[x])×fa[x]

因为还会影响到sum[fax]sum[fa_x]sum[fax​]，也要O(1)O(1)O(1)修改一下

Code：

#include<bits/stdc++.h>
using namespace std;
const int N=1e5+7;
int n,q,fa[N];
double p[N],sum[N],ans;
vector<int> son[N];
void dfs(int x,int f){
	fa[x]=f;
	for(int i=0;i<son[x].size();++i){
		int y=son[x][i];
		if(y==f)continue;
		dfs(y,x);
		sum[x]+=1-p[y];
		ans+=p[x]*(1-p[y]);
	}
}
int main(){
	scanf("%d",&n);
	p[0]=1;
	for(int i=1;i<=n;++i) scanf("%lf",p+i);
	for(int i=1;i<n;++i){
		int u,v;
		scanf("%d%d",&u,&v);
		++u,++v;
		son[u].push_back(v);
		son[v].push_back(u);
	}
	ans=1-p[1];
	dfs(1,0);
	scanf("%d",&q);
	while(q--){
		int k;double p1;
		scanf("%d%lf",&k,&p1);++k;
		sum[fa[k]]-=p1-p[k];
		ans+=sum[k]*(p1-p[k])-(p1-p[k])*p[fa[k]];
		p[k]=p1;
		printf("%.5lf\n",ans);
	}
	return 0;
}

总结：一个套路，一个Trick

CF1327F AND Segments

这题很像P4229 某位歌姬的故事，这题也是之前做过，但由于比较复杂，总结起来需要时间，所以就一直咕咕咕~，价值也失去了

本题题解看了，但不懂他的满足限制是怎么实现的

实际上很简单，只要明确每个位置pos[i]不同即可

式子：

  • j<pos[i]:f[i][j]=0j<pos[i]:f[i][j]=0j<pos[i]:f[i][j]=0

  • j<=pos[i]<i:f[i][j]=f[i−1][j]j<=pos[i]<i:f[i][j]=f[i-1][j]j<=pos[i]<i:f[i][j]=f[i−1][j]

  • j=i:f[i][j]=a[i]?0:∑k=pos[i]i−1f[i−1][k]j=i:f[i][j]=a[i]?0:\sum_{k=pos[i]}^{i-1} f[i-1][k]j=i:f[i][j]=a[i]?0:∑k=pos[i]i−1​f[i−1][k]

Code：

#include<bits/stdc++.h>
#define int long long
using namespace std;
const int N=5e5+7,mod=998244353;
int n,k,m,l[N],r[N],x[N],f[N];
int pos[N],a[N];//pos表示i之前最后一个0的位置，a表示第i个位置是否强制选1
void pre(int p){
	memset(pos,0,sizeof(pos));
	memset(a,0,sizeof(a));
	memset(f,0,sizeof(f));
	
	for(int i=1;i<=m;++i)
		if((x[i]>>p)&1) ++a[l[i]],--a[r[i]+1];
		else pos[r[i]+1]=max(pos[r[i]+1],l[i]);
	for(int i=1;i<=n+1;++i)
		a[i]+=a[i-1],pos[i]=max(pos[i],pos[i-1]);//pos也要及时更新，pos[i-1]一定合法，但不一定严格，用pos[i]补充
}
void dp(){
	f[0]=1;
	int sum=1,l=0;
	for(int i=1;i<=n+1;++i){
		while(l<pos[i]) sum=(sum-f[l]+mod)%mod,f[l]=0,++l;
		f[i]=a[i]?0:sum;
		sum=(sum+f[i])%mod;
	}
}
signed main(){
	scanf("%lld%lld%lld",&n,&k,&m);
	for(int i=1;i<=m;++i) scanf("%lld%lld%lld",l+i,r+i,x+i);
	int ans=1;
	for(int p=0;p<k;++p){
		pre(p);
		dp();
		ans=1LL*ans*f[n+1]%mod;
	}
	printf("%lld",ans);
	return 0;
}

P3523 [POI2011]DYN-Dynamite

题意：

mid一下，判断是否存在m个点能在mid的距离里到所有关键节点

将最近距离小于等于mid，转化为求需要多少个点才能到所有关键点，如果<=m<=m<=m就可行

一个点x，设f[x]f[x]f[x]为其子树内最远没有覆盖的点的距离，g[x]g[x]g[x]为最近被覆盖的点的距离，dpdpdp即可

#include<bits/stdc++.h>
using namespace std;
const int N=3e5+7,INF=1e8;
int n,m,b[N],head[N],tot;
int cnt,f[N],g[N];//f[]最远不覆盖,g[]最近被覆盖
struct E{int ver,nxt;}e[N<<1];
inline void add(int u,int v){
	e[++tot]=(E){v,head[u]},head[u]=tot;
}
void dfs(int x,int fa,int mid){
	f[x]=-INF,g[x]=INF;
	for(int i=head[x];i;i=e[i].nxt){
		int y=e[i].ver;
		if(y==fa)continue;
		dfs(y,x,mid);
		f[x]=max(f[x],f[y]+1);
		g[x]=min(g[x],g[y]+1);
	}
	if(f[x]+g[x]<=mid) f[x]=-INF;
	if(g[x]>mid&&b[x]) f[x]=max(f[x],0);
	if(f[x]==mid) f[x]=-INF,g[x]=0,++cnt;
}
bool check(int mid){
	cnt=0;
	dfs(1,0,mid);
	if(f[1]>=0)++cnt;
	return cnt<=m;
}
int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;++i) scanf("%d",b+i);
	for(int i=1;i<n;++i){
		int u,v;
		scanf("%d%d",&u,&v);
		add(u,v),add(v,u);
	}
	int l=0,r=n;
	while(l<r){
		int mid=(l+r)>>1;
		if(check(mid))r=mid;
		else l=mid+1;
	}
	printf("%d",l);
	return 0;
}

总结：“需要”多少个点是一个很好的思路，只用最大值记录子树内覆盖和未覆盖状态

P3457 [POI2007]POW-The Flood

并查集和贪心都不难想到，但排序，从小到大合并周围的点，比较巧妙！

另外是类似广搜思想的好题（就该写法），所以明天重构一遍代码

总体难度偏小

#include<bits/stdc++.h>
using namespace std;
const int N=1005;
int m,n,mp[N][N],cnt,fa[N*N],siz[N*N],ans,num[N][N];
int mov[4][2]={{1,0},{0,1},{-1,0},{0,-1}};
struct P{
	int x,y,h;
	bool operator < (const P A) const {
		return h<A.h;
	}
}p[N*N];
int find(int x){
	return fa[x]==x?x:fa[x]=find(fa[x]);
}
void Merge(int x,int y){
	x=find(x),y=find(y);
	siz[y]|=siz[x];
	fa[x]=y;
}
int main(){
	memset(mp,0x3f,sizeof(mp));
	scanf("%d%d",&m,&n);
	for(int i=1;i<=m;++i)
		for(int j=1;j<=n;++j)
			scanf("%d",&mp[i][j]);
	for(int i=1;i<=m;++i)
		for(int j=1;j<=n;++j){
			num[i][j]=cnt;
			p[++cnt]=(P){i,j,abs(mp[i][j])};
		}
	sort(p+1,p+cnt+1);
	for(int i=1;i<=cnt;++i) fa[i]=i;
	for(int i=1;i<=cnt;++i){
		int X=p[i].x,Y=p[i].y;
		for(int k=0;k<=3;++k){
			int x=X+mov[k][0],y=Y+mov[k][1];
			if(abs(mp[x][y])<=p[i].h)
				Merge(num[x][y],num[X][Y]);
		}
		if(p[i].h!=p[i+1].h)
			for(int j=i;p[j].h==p[i].h;--j)
				if(mp[p[j].x][p[j].y]>0){
					int x=find(num[p[j].x][p[j].y]);
					if(!siz[x]) ++ans,siz[x]=1;
				}
	}
	printf("%d",ans);
	return 0;
}

总结：贪心地选取最低点不一定找最低点枚举，可以先排序，再按顺序处理，提高效率

P5005 中国象棋 - 摆上马

这题状压，特点是空间很小

思路很板子，判断行列矛盾非常毒瘤！卡空间用滚动数组，都不难想，代码难写

赋初值注意一下：是前两行都赋上初值

Code：

#include<cstdio>
#include<iostream>
#define LL long long
using namespace std;
const int mod=1e9+7;
inline void inc(register int &a,register int b){
	a=a+b>=mod?a+b-mod:a+b;
}
inline int bit(register int a,register int x){
	return x<1?0:a&(1<<(x-1));
}
inline int check(register int a,register int x){
	return x<1?0:(a&(1<<(x-1))?1:0);
}
int bt1(register int a){
	int c=0;
    for(register int i=1;bit(-1,i)<=a;++i){
        if(!check(a,i))	continue;
        if(check(a,i)&check(a,i-1));
        else c|=bit(-1,i-2);
        if(check(a,i)&check(a,i+1));
        else c|=bit(-1,i+2);
    }
    return c;
}
int bt2(register int a,register int b){
	int c=0;
    for(register int i=1;bit(-1,i)<=a;++i){
        if(!check(a,i)) continue;
        if(check(a,i)&check(b,i));
        else c|=bit(-1,i-1),c|=bit(-1,i+1);
    }
    return c;
}
int f[3][1<<6][1<<6];
signed main(){
	int x,y;
	scanf("%d%d",&x,&y);
	int m=(1<<y)-1;
	for(register int s(0);s<=m;++s) f[1][s][0]=1;
	for(register int s(0);s<=m;++s)
		for(register int s1(0);s1<=m;++s1)
			if(!(bt1(s)&s1)&&!(bt1(s1)&s))
				f[2][s][s1]=f[1][s1][0];
	for(register int i(3);i<=x;++i){
		int g=i%3,k=(i-1)%3;
		for(register int s0(0);s0<=m;++s0)
            for(register int s1(0);s1<=m;++s1)
                f[g][s0][s1]=0;
		for(register int s0(0);s0<=m;++s0)
			for(register int s1(0);s1<=m;++s1)
				if(!(bt1(s0)&s1)&&!(bt1(s1)&s0))
					for(register int s2(0);s2<=m;++s2)
						if(!(bt1(s1)&s2)&&!(bt1(s2)&s1)&&!(bt2(s0,s1)&s2)&&!(bt2(s2,s1)&s0))
							inc(f[g][s0][s1],f[k][s1][s2]);
	}
	int ans=0,g=x%3;
	for(register int s(0);s<=m;++s)
		for(register int s1(0);s1<=m;++s1)
			if(!(bt1(s)&s1)&&!(bt1(s1)&s))
				inc(ans,f[g][s][s1]);
	printf("%lld",ans);
	return 0;
}
