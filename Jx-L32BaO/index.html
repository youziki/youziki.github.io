U141642 下棋（chess）

题面：

有n个白色棋子，m个黑色棋子，现在需要把他们排成一排，要求对于任意一段棋子，其中的白色棋子和黑色棋子的差不能超过k，求棋子排列方案数对1000000007(1e9+7)的结果。

注：(黑-黑-白-白) 与 (白-白-黑-黑) 视为不同的方案。

Solution：

记搜实现，令f[i][j][x][y]f[i][j][x][y]f[i][j][x][y]为已经放了i个白子，j个黑子，所有后缀中白子比黑子最多多x个，所有后缀中黑子比白子最多多y个

转移不难，关键是记搜的方向，我一开始令f[i][j][x][y]f[i][j][x][y]f[i][j][x][y]为达到这个状态的方案数，应该为“这个状态可以扩展出来的方案数”，这样边界比较好判if (i == n && j == m) return 1;，而且输出状态也容易cout << dp(0, 0, 0, 0);。考虑dfsdfsdfs不就是从初始状态开始吗？

U141645 堆石子(stone)

题面：

一堆数，合并一些数，使得最后所有堆的数%m的余数最大值最大

solution：

考场上类似小学生出的那道题的解法用bitset写的，20pts， 正解就一句话：折半

爆搜和枚举都能过

考场代码 爆搜 枚举

Code1：
#include <bits/stdc++.h>
using namespace std;
const int N = 155, K = 22, P = 1e9 + 7;
int n, m, k, f[N][N][K][K], ans;
bool vis[N][N][K][K];
int dp(int i, int j, int x, int y) {
    if (i == n && j == m) return 1;
    int &X = f[i][j][x][y];
    if (vis[i][j][x][y]) return X;
    vis[i][j][x][y] = 1;
    if (i < n && x < k) X = (X + dp(i + 1, j, x + 1, max(0, y - 1))) % P;
    if (j < m && y < k) X = (X + dp(i, j + 1, max(0, x - 1), y + 1)) % P;
    return X;
}
signed main() {
    cin >> n >> m >> k;
    cout << dp(0, 0, 0, 0);
    return 0;
}

#include <bits/stdc++.h>
using namespace std;
#define int long long
const int N = 40;
int n, m, s[N], ans;
set<int> S[2];
void dfs(int x, int l, int r) {
    x = x >= m ? x - m : x;
    if (l == r) return S[r == n + 1].insert(x), void();
    dfs(x + s[l], l + 1, r);
    dfs(x, l + 1, r);
}
signed main() {
    cin >> n >> m;
    for (int i = 1; i <= n; ++i) {cin >> s[i]; s[i] %= m;}
    dfs(0, 1, n / 2 + 1); S[0].insert(0);
    dfs(0, n / 2 + 1, n + 1); S[1].insert(0);
    for (set<int>::iterator it = S[0].begin(); it != S[0].end(); ++ it) {
        int x = m - 1 - *it;
        ans = max(ans, *(-- S[1].upper_bound(x)) + *it);
    }
    cout << ans;
    return 0;
}
