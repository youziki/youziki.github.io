小清新构造题 （这么简单，考场上居然没有想出来。。。）

不可能输出-1/lh/lh/lh

一看就懂系列，上图：

这是基本框架

这是构造方法

Code：

#include <bits/stdc++.h>
using namespace std;

const int N = 505;

int n, m;
char mp[N][N];

signed main()
{
	scanf("%d%d", &n, &m);
	for (int i = 1; i <= n; ++i) scanf("%s", mp[i] + 1);
	
	for (int i = 1; i <= n; ++i, putchar('\n'))
		for (int j = 1; j <= m; ++j)
			putchar( (j == 1 || (i & 1) || mp[i][j] == '1') && j != m ? '1' : '0');
	
	putchar('\n');
	
	for (int i = 1; i <= n; ++i, putchar('\n'))
		for (int j = 1; j <= m; ++j)
			putchar( (j == m || ! (i & 1) || mp[i][j] == '1') && j != 1 ? '1' : '0');
	
	return 0;
}
