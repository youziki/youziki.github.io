一看就懂系列，上图：

这是基本框架

这是构造方法

Code：

#include <bits/stdc++.h>
using namespace std;

const int N = 505;

int n, m;
char mp[N][N];

signed main()
{
	scanf("%d%d", &n, &m);
	for (int i = 1; i <= n; ++i) scanf("%s", mp[i] + 1);
	
	for (int i = 1; i <= n; ++i, putchar('\n'))
		for (int j = 1; j <= m; ++j)
			putchar( (j == 1 || (i & 1) || mp[i][j] == '1') && j != m ? '1' : '0');
	
	putchar('\n');
	
	for (int i = 1; i <= n; ++i, putchar('\n'))
		for (int j = 1; j <= m; ++j)
			putchar( (j == m || ! (i & 1) || mp[i][j] == '1') && j != 1 ? '1' : '0');
	
	return 0;
}

T3 C

Solution：

不太懂边界

Code：

int ans = 0;
for (re int i = 0; i < y; i++)
{
  l = 1 - l;
  memset(f[1 - l], 0, sizeof f[1 - l]);

  for (re int r = 0; r <= m - y; r++)
    for (re int j = 0; j < x; j++)
      for (re int z = 0; z <= n - x; z++)
      {
        ans = std::max(ans, f[l][r][j][z]);

        if (i + r < y - 1)
          f[1 - l][r][j][z] = max(f[1 - l][r][j][z],
					  		f[l][r][j][z]
							  + column[y - i - 1][std::min(n - j, x + z)]
							  - column[y - i - 1][std::max(z, x - j - 1)]);
        else
          f[1 - l][r][j][z] = max(f[1 - l][r][j][z],
		  					f[l][r][j][z]);
            
        if (i + r < (m - y))
          f[l][r + 1][j][z] = max(f[l][r + 1][j][z],
		  					f[l][r][j][z]
							  + column[y + r + 1][std::min(n - j, x + z)]
							  - column[y + r + 1][std::max(z, x - j - 1)]);
        else
          f[l][r + 1][j][z] = max(f[l][r + 1][j][z],
		  					f[l][r][j][z]);
            
		if (j + z < x - 1)
          f[l][r][j + 1][z] = max(f[l][r][j + 1][z],
		  					f[l][r][j][z]
							  + line[x - j - 1][std::min(m - i, y + r)]
							  - line[x - j - 1][std::max(r, y - i - 1)]);
     	else
          f[l][r][j + 1][z] = max(f[l][r][j + 1][z],
		  					f[l][r][j][z]);
            
		if (n - j > z + x)
          f[l][r][j][z + 1] = max(f[l][r][j][z + 1],
		  					f[l][r][j][z]
							  + line[z + x + 1][std::min(y + r, m - i)]
							  - line[z + x + 1][std::max(r, y - i - 1)]);
        else
          f[l][r][j][z + 1] = max(f[l][r][j][z + 1],
		  					f[l][r][j][z]);
      }
}

T5 萌猫后卫

看到数据范围应该想到区间dp！有一个坑点，很好，需要记住！

我在枚举划分点的时候枚举的是先干掉哪个人，这样是错的，因为干掉这个人之后，两边的区间会合并，不能变成独立的两部分。

正确做法应该是枚举最后干掉哪个人，这样，干掉这个人之前，两边一定是独立的两部分。而且这样做计算代价也很简单，因为划分点两边永远有人（k=1或k=n的两边的b[]值刚好为0，不用特判）

Code：

int dp(int i, int j)
{
	if (i > j) return 0;
	if (vis[i][j]) return f[i][j];
	vis[i][j] = 1;
	
	for (int k = i; k <= j; ++k)
		f[i][j] = min(f[i][j],
				dp(i, k - 1) + dp(k + 1, j) + (h[k] + at - 1) / at * (a[k] + b[i - 1] + b[j + 1]));
	
	return f[i][j];
}
