T4 circular

会了O(n2)O(n^2)O(n2)做法，O(nlogn)O(nlogn)O(nlogn)做法就是一句话的事儿了，先看前者

倍长成链、枚举位置不难想，关键是如何处理贪心的部分（选最多不相交的区间），贪心策略也不难想：选择一个区间，下一个区间应该为“左端点在这个区间右端点或以后且右端点最靠左的区间”，如何O(n)O(n)O(n)找出这部分？

在每次贪心和预处理中选择预处理

用一个nxtnxtnxt数组记录一个区间选择之后的后继，将所有端点排序，用一个lstlstlst指针从后往前扫，时刻记录扫到的最靠前的sss端点，遇到ttt就附lstlstlst即可，具体看代码

如何优化到O(nlogn)O(nlogn)O(nlogn)?把一步一步跳nxtnxtnxt替换成倍增，预处理个ststst表即可

init
const int N = 200005;

int n, m, mx, s[N], t[N], nxt[N][23];
pii b[N << 1];

m = read(), n = read();
for(int i = 1; i <= n; ++i)
{
    s[i] = read(), t[i] = read();
    if(s[i] > t[i]) t[i] += m;
    s[n + i] = s[i] + m;
    t[n + i] = t[i] + m;
}
get_nxt
mx = n << 1;
for(int i = 1; i <= mx; ++i)
{
    b[i] = m_p(t[i], i);
    b[i + mx] = m_p(s[i], i + mx);
    // 这里实际上就是加入所有端点，第二维就是为了找到编号方便
}

sort(b + 1, b + (mx << 1) + 1);
int lst = -1;
for(int i = (mx << 1); i; --i)
{
    int id = b[i].se;
    if(id <= mx) // 如果是t，就赋值
        nxt[id][0] = lst;
    else
    {
        id -= mx;
        if(lst < 0 || t[lst] > t[id]) // 更新lst
            lst = id;
    }
}
st_pre
for(int i = 1; i <= 20; ++i)
    for(int j = 1; j + (1 << i) <= mx; ++j)
        if(nxt[j][i - 1] == -1)
            nxt[j][i] = -1;
        else
            nxt[j][i] = nxt[ nxt[j][i - 1] ][i - 1];
work
for(int i = 1; i <= n; ++i)
{
    int sum = 0, j = i;
    for(int k = 20; ~k; --k)
    {
        int _j = nxt[j][k];
        if(_j > 0 && t[_j] <= s[i] + m)
        {
            j = _j;
            sum |= 1 << k;
        }
    }
    ans = max(ans, sum + 1);
}
